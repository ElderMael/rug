51   /**
52     * Return the atomist content only
53     *
54     * @param rugArchive artifact source
55     * @return only the Atomist content from the archive
56     */
57   def atomistContent(rugArchive: ArtifactSource): ArtifactSource = {
58 
59     //find files in dir (if any)
60     def files(dir: String): Seq[FileArtifact] = {
61       rugArchive.findDirectory(dir) match {
62         case Some(found) => found.allFiles
63         case _ => Nil
64       }
65     }
66 
67     val atomistFiles = files(editorsRoot) ++ files(generatorsRoot) ++ files(handlersRoot) ++ files(testsRoot)
68     ArtifactSource.fromFiles(atomistFiles: _*)
69   }
70 
71 
72   def isJsSource(f: FileArtifact): Boolean = {
73     f.name.endsWith(jsExtension) && isAtomistSource(f)
74   }
75 
76   def isJsTest(f: FileArtifact): Boolean = {
77     // TODO fix hard coding
78     f.name.endsWith(jsExtension) && f.path.startsWith(s"$atomistRoot/test")
79   }
80 
81   def isAtomistSource(f: FileArtifact): Boolean = {
82     f.path.startsWith(editorsRoot) ||
83       f.path.startsWith(generatorsRoot) ||
84       f.path.startsWith(handlersRoot) ||
85       f.path.startsWith(handlersDirectory) ||
86       f.path.startsWith(editorsDirectory) ||
87       f.path.startsWith(generatorsDirectory)
88   }
89 
90   def isJsHandler(f: FileArtifact): Boolean = {
91     f.name.endsWith(jsExtension) && (f.path.startsWith(handlersRoot) ||
92       f.path.startsWith(handlersDirectory))
93   }
94 
95   def templateContentIn(rugAs: ArtifactSource): ArtifactSource =
96     rugAs / templatesRoot + rugAs / templatesDirectory
97 
98 }
