9 /**
10   * Used by PlanRunner and Project Operations to find Rugs
11   *
12   * We explicitly do _not_ support loading a Rug that is not a direct dependency
13   * i.e. you cannot load a Rug that is brought in transitively
14   *
15   * A None means no resolution!
16   */
17 class ArchiveRugResolver(graph: Dependency,
18                          engineInitializer: EngineInitializer = redirectConsoleToSysOut) extends RugResolver {
19 
20   private val finders: Seq[JavaScriptRugFinder[_ <: Rug]] = Seq(
21     new JavaScriptCommandHandlerFinder(),
22     new JavaScriptResponseHandlerFinder(),
23     new JavaScriptEventHandlerFinder(),
24     new JavaScriptProjectGeneratorFinder(),
25     new JavaScriptProjectEditorFinder())
26 
27   /**
28     * Find Rugs of all known kinds in the artifact.
29     */
30   private def find(as: ArtifactSource, resolver: RugResolver): Seq[Rug] = {
31     val jsc = new JavaScriptContext(as, initializer = engineInitializer)
32     val found = finders.flatMap(finder => finder.find(jsc, Some(resolver)))
33     val grouped = found.groupBy(_.name).collect{case x if x._2.size > 1 => x._1}
34     if(grouped.nonEmpty){
35       throw new DuplicateRugException(s"The following rugs have duplicates in the archive: ${grouped.mkString(", ")}", found)
36     }else{
37       found
38     }
39   }
40 
41   //group:artifact:name
42   private def fqRex = "^(.*?):(.*?):(.*?)$".r
43 
44   /**
45     * Top of the dep graph.
46     */
47   override val resolvedDependencies: ResolvedDependency = {
48     val rugs = find(graph.source, this)
49     ResolvedDependency(graph.source, graph.address, dependencies(graph.dependencies), rugs)
50   }
51 
52   /**
53     * Recursively resolve all the rugs.
54     */
55   private def dependencies(deps: Seq[Dependency]): Seq[ResolvedDependency] = {
56     deps.map(dep =>
57       if (dep.dependencies.nonEmpty) {
58         ResolvedDependency(dep.source, dep.address, dependencies(dep.dependencies), find(dep.source, this))
59       } else {
60         ResolvedDependency(dep.source, dep.address, Nil, find(dep.source, this))
61       }
62     )
63   }
64 
65   /**
66     * Search direct dependencies of root for a rug.
67     */
68   override def resolve(root: Rug, nameOrFqName: String): Option[Rug] = {
69     findResolvedDependency(root) match {
70       case Some(rootDep) =>
71         val local = rootDep.resolvedRugs.find(p => rugMatches(p, rootDep, nameOrFqName))
72         if (local.nonEmpty) {
73           local
74         } else {
75           (for {
76             dep <- rootDep.dependencies
77             rug <- dep.resolvedRugs
78             if rugMatches(rug, dep, nameOrFqName)
79           } yield rug).headOption
80         }
81       case _ => None
82     }
83   }
84 
85   /**
86     * Does the current rug/dep match the name?
87     */
88   private def rugMatches(rug: Rug, rootDep: ResolvedDependency, name: String): Boolean = {
89     name match {
90       case simple: String if !simple.contains(":") => simple == rug.name
91       case fq: String => (fqRex.findFirstMatchIn(fq), rootDep.address) match {
92         case (Some(m), Some(address)) =>
93           m.group(1) == address.group &&
94             m.group(2) == address.artifact &&
95             m.group(3) == rug.name
96         case _ => false
97       }
98     }
99   }
100 
101   /**
102     * Find the node in the dependency graph for a given Rug instance.
103     */
104   override def findResolvedDependency(rug: Rug): Option[ResolvedDependency] = {
105     findRug(rug, resolvedDependencies)
106   }
107 
108   /**
109     * Find the root rug in the graph.
110     */
111   private def findRug(root: Rug, resolved: ResolvedDependency): Option[ResolvedDependency] = {
112     resolved.resolvedRugs.find(r => r == root) match {
113       case Some(_) => Some(resolved)
114       case _ =>
115         val matched = resolved.dependencies.map(findRug(root, _))
116         if (matched.isEmpty) {
117           None
118         }
119         else {
120           matched.head
121         }
122     }
123   }
