171   /** If obj is a String:
172     * -  If there are allowedValues, return true if value is in the list of allowedValues
173     * -  If allowedValues is empty, return true if value meets the minimum and maximum
174     *    length requirements and it satisfies the parameter validation regular
175     *    expression.
176     * Otherwise return false.
177     * If obj is not a String, return false.
178     *
179     * @param obj value to be checked
180     * @return true is the obj is a valid value, false otherwise.
181     */
182   def isValidValue(obj: Any): Boolean = obj match {
183     case s: String =>
184       if (allowedValues.nonEmpty) allowedValues.exists(_.value == s)
185       else ((minLength < 0 || s.length >= minLength) &&
186         (maxLength < 0 || s.length <= maxLength) &&
187         pattern.r.findAllMatchIn(s).nonEmpty)
188     case _ => false
189   }
