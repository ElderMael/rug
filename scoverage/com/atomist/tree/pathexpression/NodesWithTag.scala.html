25   private def childResolver(typeRegistry: TypeRegistry, gn: GraphNode): Option[ChildResolver] =
26     typeRegistry.findByName(tag) match {
27       case Some(cr: ChildResolver) =>
28         Some(cr)
29       case Some(_) =>
30         // Type is found, but not a ChildResolver. Not an error
31         None
32       case None if tag.charAt(0).isLower =>
33         // It starts with lower case. Give it the benefit of the doubt.
34         // It won't be a microgrammar reference anymore ... but maybe an AST node that might exist but doesn't?
35         None
36       case _ =>
37         // With dynamic nodes we can't know this is invalid. A TreeMaterializer backed path expression
38         // that doesn't match and thus had no kids is a valid case and it won't be able to define types
39         if (gn.hasTag(TreeNode.Dynamic)) {
40           None
41         }
42         else {
43           // Type is unknown. This is an error
44           throw NoSuchTypeException
45         }
46     }

48   private val eligibleNode: GraphNode => Boolean =
49     n => n.hasTag(tag) || n.nodeName == tag

51   // Attempt to find nodes of the required type under the given node
52   private def findMeUnder(gn: GraphNode, ec: ExecutionContext, edgeName: Option[String]): Seq[GraphNode] = {
53     // Filter for node name if we're following an edge
54     val nameFilter: GraphNode => Boolean = edgeName match {
55       case None | Some("*") => _ => true
56       case Some(name) => n => n.nodeName == name
57     }
58     val r = gn.relatedNodes.filter(eligibleNode).filter(nameFilter) match {
59       case Nil =>
60         childResolver(ec.typeRegistry, gn) match {
61           case Some(cr) => edgeName match {
62             case None =>
63               cr.findAllIn(gn).getOrElse(Nil)
64             case Some(name) =>
65               cr.navigate(gn, name, ec).getOrElse(Nil)
66           }
67           case None =>
68             Nil
69         }
70       case kids =>
71         kids
72     }
73     logger.debug(s"$this Returned $r for edgeName=$edgeName below $gn")
74     r
75   }

77   override def follow(gn: GraphNode, axis: AxisSpecifier,
78                       ee: ExpressionEngine,
79                       executionContext: ExecutionContext): ExecutionResult = {
80     try {
81       axis match {
82         case NavigationAxis(propertyName) =>
83           // Follow edge or look for nodes with the name
84           val nodes = gn.followEdge(propertyName).filter(eligibleNode)
85           if (nodes.nonEmpty)
86             ExecutionResult(nodes)
87           else {
88             ExecutionResult(gn.relatedNodesNamed(propertyName).filter(eligibleNode)) match {
89               case Right(Nil) =>
90                 ExecutionResult(findMeUnder(gn, executionContext, Some(propertyName)))
91               case l => l
92             }
93           }
94         case Self =>
95           ExecutionResult(List(gn).filter(eligibleNode))
96         case Child =>
97           ExecutionResult(findMeUnder(gn, executionContext, None))
98         case Descendant =>
99           val allDescendants = Descendant.selfAndAllDescendants(gn)
100           val found: Seq[GraphNode] = allDescendants.flatMap(d =>
101             findMeUnder(d, executionContext, None))
102           ExecutionResult(found)
103         case x => throw new UnsupportedOperationException(s"Unsupported axis $x in ${getClass.getSimpleName}")
104       }
105     }
106     catch {
107       case NoSuchTypeException =>
108         Left(s"No such type: [$tag]")
109     }
110   }
