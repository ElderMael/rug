14 
15   /**
16     * We allow - and _ in navigation axis, as they are valid in
17     * GitHub branch names, and . as it may be used in files
18     */
19   private def legalNavigationOrChild: Parser[String] =
20     """[\.a-zA-Z0-9_\-\$#]+""".r
21 
22   private def nodeName: Parser[String] = identifierRefString(Set.empty, legalNavigationOrChild)
23 
24   private def functionName: Parser[String] = identifierRefString(Set.empty, "[a-zA-Z][a-zA-Z0-9\\-]+".r)
25 
26   private def objectType: Parser[String] = identifierRefString(Set.empty, ident)
27 
28   private def child: Parser[AxisSpecifier] = opt(s"$ChildAxis::") ^^ {
29     case Some(_) => Child
30     case _ => Child
31   }
32 
33   private def navigationAxis: Parser[AxisSpecifier] =
34     identifierRefString(StandardAxes, legalNavigationOrChild) <~ "::" ^^
35       (s => NavigationAxis(s))
36 
37   private def nodeTypeTest: Parser[NodesWithTag] = objectType <~ "()" ^^ (p => NodesWithTag(p))
38 
39   private def propertyTest: Parser[Predicate] =
40     "@" ~> nodeName ~ EqualsToken ~ (singleQuotedString | doubleQuotedString) ^^ {
41       case prop ~ op ~ literal => prop match {
42         case "name" =>
43           NodeNamePredicate(literal)
44         case "type" =>
45           NodeTypePredicate(literal)
46         case propName: String =>
47           PropertyValuePredicate(propName, literal)
48       }
49     }
50 
51   private def nullLiteral: Parser[Object] = "null" ^^ (_ => null)
52 
53   private def integer: Parser[Integer] = decimalNumber ^^ (s => s.toInt)
54 
55   private def literal: Parser[Any] = nullLiteral | singleQuotedString | integer
56 
57   private def methodInvocationTest: Parser[Predicate] = "." ~> nodeName ~ args ~ EqualsToken ~ literal ^^ {
58     case methodName ~ args ~ _ ~ literal =>
59       FunctionPredicate(s".$methodName", (n, among) => {
60         val rO = invokeMethodIfPresent[Any](n, methodName, args)
61         rO.exists(invoked => Objects.equals(literal, invoked))
62       })
63   }
64 
65   private def arg: Parser[String] = singleQuotedString
66 
67   private def args: Parser[Seq[String]] = "(" ~> repsep(arg, ",") <~ ")"
68 
69   private def stringLiteralFunctionArg: Parser[StringLiteralFunctionArg] =
70     (singleQuotedString | doubleQuotedString) ^^ (
71       s => StringLiteralFunctionArg(s)
72       )
73 
74   private def relativePathFunctionArg: Parser[RelativePathFunctionArg] =
75     nonEmptyRelativePathExpression ^^ (
76       s => RelativePathFunctionArg(s)
77       )
78 
79   private def functionArg: Parser[FunctionArg] =
80     stringLiteralFunctionArg |
81       relativePathFunctionArg
82 
83   private def functionArgs: Parser[Seq[FunctionArg]] = "(" ~> rep1sep(functionArg, ",") <~ ")"
84 
85   private def booleanMethodInvocation: Parser[Predicate] = "." ~> functionName ~ args ^^ {
86     case methodName ~ args =>
87       FunctionPredicate(s".$methodName", (n, among) =>
88         invokeMethodIfPresent[Boolean](n, methodName, args).getOrElse(false)
89       )
90   }
91 
92   private def functionCall: Parser[Predicate] = functionName ~ functionArgs ^^ {
93     case functionName ~ args =>
94       XPathStyleFunctionPredicate(functionName, args)
95   }
96 
97   private def index: Parser[Predicate] = integer ^^ {
98     n => IndexPredicate(n)
99   }
100 
101   private def truePredicate: Parser[Predicate] = "true" ^^ (_ => TruePredicate)
102 
103   private def falsePredicate: Parser[Predicate] = "false" ^^ (_ => FalsePredicate)
104 
105   private def nestedPathExpressionPredicate: Parser[Predicate] =
106     pathExpression ^^ (pe => NestedPathExpressionPredicate(pe))
107 
108   private def predicateTerm: Parser[Predicate] =
109     methodInvocationTest |
110       propertyTest |
111       booleanMethodInvocation |
112       truePredicate |
113       falsePredicate |
114       functionCall |
115       index |
116       nestedPathExpressionPredicate
117 
118   private def negatedPredicate: Parser[Predicate] = "not" ~> "(" ~> predicateExpression <~ ")" ^^ {
119     pred => NegationOfPredicate(pred)
120   }
121 
122   private def logicalOp: Parser[String] = "and" | "or"
123 
124   private def predicateAnd: Parser[Predicate] = predicateTerm ~ logicalOp ~ predicateExpression ^^ {
125     case a ~ "and" ~ b => a and b
126     case a ~ "or" ~ b => a or b
127   }
128 
129   private def predicateExpression: Parser[Predicate] = predicateAnd | negatedPredicate | predicateTerm
130 
131   private def predicate: Parser[Predicate] =
132     PredicateOpen ~> predicateExpression ~ PredicateClose ~ opt(PredicateOptional) ^^ {
133       case p ~ _ ~ Some(_) => OptionalPredicate(p)
134       case p ~ _ ~ None => p
135     }
136 
137   private def nodeNameTest: Parser[NodeTest] = nodeName ^^
138     (s => NamedNodeTest(s))
139 
140   private def allNodes: Parser[NodeTest] = "*" ^^ (_ => All)
141 
142   private def nodeTest: Parser[NodeTest] = nodeTypeTest | nodeNameTest | allNodes
143 
144   private def property: Parser[AxisSpecifier] = (s"$PropertyAxis::" | "@") ^^ (s => Attribute)
145 
146   private def descendant: Parser[AxisSpecifier] = (s"$DescendantAxis::" | "/") ^^
147     (s => Descendant)
148 
149   private def selfAxis: Parser[AxisSpecifier] = (s"$SelfAxis::" | ".") ^^
150     (s => Self)
151 
152   private def axis: Parser[AxisSpecifier] =
153     property |
154       navigationAxis |
155       descendant |
156       selfAxis |
157       child
158 
159   private def locationStep: Parser[LocationStep] = axis ~ nodeTest ~ rep(predicate) ^^ {
160     case a ~ t ~ preds => LocationStep(a, t, preds)
161   }
162 
163   private val slashSeparator = "/"
164 
165   def relativePathExpression: Parser[PathExpression] = (selfAxis | repsep(locationStep, slashSeparator)) ^^ {
166     case Self => PathExpression(List(LocationStep(Self, All, Nil)))
167     case steps: List[LocationStep]@unchecked => PathExpression(steps)
168   }
169 
170   private def nonEmptyRelativePathExpression: Parser[PathExpression] = new Parser[PathExpression] {
171 
172     override def apply(in: Input): ParseResult[PathExpression] = {
173       relativePathExpression(in) match {
174         case s: Success[PathExpression]@unchecked if s.result.locationSteps.nonEmpty => s
175         case s: Success[PathExpression]@unchecked => Failure(s"Path expression has no steps: [${s.result}]", in)
176         case x => x
177       }
178     }
179   }
180 
181   def absolutePathExpression: Parser[PathExpression] = slashSeparator ~> relativePathExpression
182 
183   def pathExpression: Parser[PathExpression] = absolutePathExpression | relativePathExpression
184 
185   def parsePathExpression(expr: String): PathExpression = {
186     try {
187       parseTo(StringFileArtifact("<input>", expr), phrase(absolutePathExpression))
188     } catch {
189       case e: IllegalArgumentException =>
190         throw new IllegalArgumentException(s"Path expression '$expr' is invalid: [${e.getMessage}]", e)
191     }
192   }
193 }
