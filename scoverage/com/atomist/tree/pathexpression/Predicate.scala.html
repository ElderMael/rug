11 /**
12   * Based on the XPath concept of a predicate. A predicate acts on a sequence of nodes
13   * returned from navigation to filter them.
14   * Predicates can be evaluated against materialized objects, and most predicates expose enough information
15   * to generate queries against external systems to retrieve data.
16   */
17 trait Predicate {
18 
19   @JsonProperty
20   def name: String = getClass.getSimpleName.replace("$", "")
21 
22   /**
23     * Function taking nodes returned by navigation
24     * to filter them. We test one node with knowledge of all returned nodes.
25     *
26     * @param nodeToTest    node we're testing on;
27     * @param returnedNodes all nodes returned. This argument is
28     *                      often ignored, but can be used to discern the index of the target node.
29     */
30   def evaluate(nodeToTest: GraphNode,
31                returnedNodes: Seq[GraphNode],
32                ee: ExpressionEngine,
33                executionContext: ExecutionContext,
34                nodePreparer: Option[NodePreparer]): Boolean
35 
36   def and(that: Predicate): Predicate =
37     AndPredicate(this, that)
38 
39   def or(that: Predicate): Predicate =
40     OrPredicate(this, that)
41 
42   def not: Predicate =
43     NegationOfPredicate(this)
44 
45 }
