6 /**
7   * Uses reflection to load functions from designated classes
8   */
9 class ReflectiveFunctionRegistry(targets: Object*) extends FunctionRegistry {
10 
11   override val functions: Seq[Function] = {
12     for {
13       t <- targets
14       m <- t.getClass.getMethods
15       if m.getParameterCount >= 1
16       if m.getParameterTypes.forall(c => c == classOf[String] || c == classOf[Boolean])
17     } yield {
18       val declaredArgs = m.getParameterTypes.map(_.getSimpleName) map {
19         case "String" => String
20         case "Boolean" => Boolean
21         case wtf => throw new IllegalStateException(s"Should have filtered out parameter type [$wtf]")
22       }
23       // XPath functions are of the form string-length: We need to convert from Java camel case
24       SimpleFunction(JavaHelpers.toLowerCaseDelimited(m.getName, "-"),
25         declaredArgs,
26         invoker = args => {
27           //println(s"Invoking $m on $DefaultFunctions with args=${args.map(a => s"$a - ${a.getClass}")}")
28           val objectified = args map {
29             case o: Object => o
30             case _ => ???
31           }
32           m.invoke(t, objectified: _*)
33         })
34     }
35   }
36 
37   /**
38     * Note that we don't need to write defensive code here as there will be the correct number of arguments
39     * and they will have been coerced as to type
40     */
41   private case class SimpleFunction(name: String, argTypes: Seq[XPathType], invoker: Seq[Any] => Any) extends Function {
42 
43     //println(s"Creating $this")
44 
45     override def invoke(convertedArgs: Seq[Any]): Any = {
46       require(convertedArgs.size == argTypes.size)

48     }
49   }
