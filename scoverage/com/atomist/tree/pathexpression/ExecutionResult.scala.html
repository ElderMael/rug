26   /**
27     * Deduplicate a sequence of results
28     * We may have duplicates in the found collection because, for example,
29     * we might find the Java() node SomeClass.java under the directory "/src"
30     * and under "/src/main" and under the actual file.
31     * So check that our returned types have distinct addresses
32     */
33   private def dedupe(results: Seq[GraphNode]): Seq[GraphNode] = {
34     var nodeAddressesSeen: Set[String] = Set()
35     var toReturn = List.empty[GraphNode]
36     results.distinct.foreach {
37       case gn: AddressableGraphNode if gn.address != null =>
38         if (!nodeAddressesSeen.contains(gn.address)) {
39           nodeAddressesSeen = nodeAddressesSeen + gn.address
40           toReturn = toReturn :+ gn
41         }
42       case n =>
43         // There's no concept of an address here, so we have to take on trust that
44         // we didn't find this thing > once
45         toReturn = toReturn :+ n
46     }
47     toReturn
48   }
