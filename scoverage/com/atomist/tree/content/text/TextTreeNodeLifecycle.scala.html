8   /**
9     * This is how tree nodes progress:
10     *
11     * ┌───────────────────┐                    ┌───────────────────┐
12     * │                   │                    │                   │
13     * ───parsing────▶│PositionedTreeNode │─────makeReady─────▶│ UpdatableTreeNode │───────▶  proceed with
14     * │                   │                    │                   │            updates
15     * └───────────────────┘                    └───────────────────┘
16     *
17     * Parsing produces PositionedTreeNodes, which know their offset.
18     * makeReady incorporates the original contents of the file, and also the file's MutableView,
19     * so that the output UpdatableTreeNodes can report their value, accept a new value,
20     * and propagate that value all the way back to the file.
21     */
22 
23   /* This is for Antlr grammars, or anything that represents the entire file in a single parsed node. */
24   def makeWholeFileNodeReady(typeName: String,
25                              parsed: PositionedTreeNode,
26                              fileArtifact: FileArtifactBackedMutableView,
27                              preprocess: String => String = identity,
28                              postprocess: String => String = identity): UpdatableTreeNode = {
29     val inputAsParsed = preprocess(fileArtifact.content)
30     val parsedWithWholeFileOffsets =
31       ImmutablePositionedTreeNode(parsed).copy(
32         startPosition = OffsetInputPosition(0),
33         endPosition = OffsetInputPosition(inputAsParsed.length))
34     makeReady(typeName, Seq(parsedWithWholeFileOffsets), fileArtifact, preprocess, postprocess).head
35   }
36 
37   /**
38     * Some or all of the file has been parsed into some PositionedTreeNodes.
39     * (Since microgrammars aren't here anymore, it should always be ALL afaik, unless the
40     * language extension chooses to return a node or nodes representing only part of the file.
41     * It could happen.)
42     *
43     * Before we can update any of the nodes or their children, we have to:
44     *
45     * Wrap these in one node that always contains the entire content of the file; fill
46     * any unrepresented characters with padding.
47     * Transform the PositionedTreeNodes into UpdatableTreeNodes.
48     * And link that top-level node to the file, so that updates to any node propagate to the file.
49     * Cascade parentage information down into all the nodes.
50     * Return the new, Updatable representations of the input PositionedTreeNodes.
51     */
52   def makeReady(typeName: String,
53                 matches: Seq[PositionedTreeNode],
54                 fileArtifact: FileArtifactBackedMutableView,
55                 preprocess: String => String = identity,
56                 postprocess: String => String = identity): Seq[UpdatableTreeNode] = {
57     val inputAsParsed = preprocess(fileArtifact.content)
58 
59     val ottn: OverwritableTextTreeNode =
60       ImmutablePositionedTreeNode.pad(typeName: String, matches, inputAsParsed)
61 
62     val wrapperNodeContainingWholeFileContent =
63       new OverwritableTextInFile(typeName, ottn.allKidsIncludingPadding, postprocess)
64     wrapperNodeContainingWholeFileContent.setParent(fileArtifact)
65     wrapperNodeContainingWholeFileContent.childNodes.collect {
66       case utn: UpdatableTreeNode => utn // should be all of them
67       case _ => ???
68     }
69   }
