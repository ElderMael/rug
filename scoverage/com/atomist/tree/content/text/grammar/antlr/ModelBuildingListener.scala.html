56   // Use reflection to extract information from the generated methods and fields in this class
57   private def treeToContainerField(rc: ParserRuleContext): AntlrPositionedTreeNode = {
58     val rule = this.getRuleByKey(rc.getRuleIndex)
59 
60     if (rc.exception != null) throw rc.exception
61 
62     val startPos = // position(rc.getStart)
63       OffsetInputPosition(rc.getStart.getStartIndex)
64     val endPos = // position(rc.stop) + rc.getStop.getText.size
65       OffsetInputPosition(rc.getStop.getStopIndex + 1)
66 
67     // We only want methods on the generated class itself
68     val valueMethods = rc.getClass
69       .getDeclaredMethods
70       .filter(m => !Excludes.ExcludedMethods.contains(m.getName))
71 
72     val fieldsToAdd = ListBuffer.empty[AntlrPositionedTreeNode]
73 
74     def addField(f: AntlrPositionedTreeNode): Unit = f match {
75       case ptn: PositionedTreeNode =>
76         if (!fieldsToAdd.exists {
77           case p: PositionedTreeNode if ptn.hasSamePositionAs(p) => true
78           case _ => false
79         })
80           fieldsToAdd.append(f)
81       case tn =>
82         fieldsToAdd.append(tn)
83     }
84 
85     for {
86       m <- valueMethods
87       if m.getParameterCount == 0
88     } {
89       val value = m.invoke(rc)
90       for (f <- makeField(m.getName, value))
91         addField(f)
92     }
93 
94     for (f <- rc.getClass.getDeclaredFields) {
95       val value = f.get(rc)
96       for (mf <- makeField(f.getName, value))
97         addField(mf)
98     }
99 
100     val deduped = deduplicate(fieldsToAdd)
101     val namey = namingStrategy.nameForContainer(rule, deduped)
102     AntlrPositionedTreeNode.parent(
103       namey,
104       startPos,
105       endPos,
106       deduped,
107       namingStrategy.tagsForContainer(rule, deduped) ++ Set(TreeNode.Dynamic),
108       namingStrategy.significance(rule, deduped))
109   }

111   // Remove duplicate fields. The ones with lower case can replace the ones with upper case
112   // For example, handle case method_name=IDENTIFIER
113   private def deduplicate(fields: Seq[AntlrPositionedTreeNode]): Seq[AntlrPositionedTreeNode] = {
114     val lexerPred: TreeNode => Boolean = f => f.nodeName.charAt(0).isUpper
115     val lexerFields = fields.filter(lexerPred)
116 
117     val positionedParserFields = fields
118       .filter(!lexerPred(_)) collect {
119       case pv: PositionedTreeNode => pv
120     }
121 
122     // There should be no overlap between these collections
123     require(!lexerFields.exists(positionedParserFields.contains(_)))
124 
125     def theresAnAliasFieldWithProbablySamePosition(what: TreeNode) = what match {
126       case pf: PositionedTreeNode =>
127         positionedParserFields.exists(ppf => pf.hasSamePositionAs(ppf))
128       case _ => false
129     }
130 
131     // We don't want lexer fields that have an alias field with the same position
132     val deduped = fields.filter(f => !(lexerFields.contains(f) &&
133       theresAnAliasFieldWithProbablySamePosition(f))
134     )
135 
136     def unwantedDuplicates(fields: Seq[AntlrPositionedTreeNode]): Seq[AntlrPositionedTreeNode] =
137       if (fields.size <= 1) Nil
138       else fields.head match {
139         case pm: PositionedTreeNode =>
140           fields.tail.collect {
141             case pm2: PositionedTreeNode if pm.hasSamePositionAs(pm2) =>
142               pm2
143           }
144         case _ => unwantedDuplicates(fields.tail)
145       }
146 
147     val unwanted = unwantedDuplicates(deduped)
148     val filtered = deduped.filterNot(unwanted.contains(_))
149     require(unwantedDuplicates(filtered).isEmpty)
150     filtered
151   }

153   private def position(tok: Token): InputPosition = {
154     val input = tok.getInputStream.toString
155     // Antlr indexes lines from 0 and columns from 1.
156     val pos = tok.getStartIndex match {
157       case -1 => LineInputPositionImpl(input, tok.getLine, tok.getCharPositionInLine + 1)
158       case n => OffsetInputPosition(n)
159     }
160     pos
161   }

163   private def makeField(name: String, value: Object): Seq[AntlrPositionedTreeNode] = {
164     val r = value match {
165       case en: ErrorNode =>
166         logger.info(s"ErrorNode: $name=$en")
167         Seq() // does this have position information?
168       case ct: Token if ct.getText.startsWith(s"<missing ") =>
169         // Handle Antlr empty values
170         val sf = AntlrPositionedTreeNode.leaf(name, position(ct), "", Set(name), TreeNode.Signal)
171         Seq(sf)
172       case ct: Token if ct.getText == "<EOF>" =>
173         // For some reason, some grammars put the EOF on the token input stream
174         Seq()
175       case ct: Token =>
176         val pos = position(ct)
177         val text = ct.getText
178         val namey = namingStrategy.nameForTerminal(name, text)
179         val sf = AntlrPositionedTreeNode.leaf(
180           namey,
181           pos,
182           text,
183           Set(name), TreeNode.Signal)
184         Seq(sf)
185       case tn: TerminalNode =>
186         makeField(name, tn.getSymbol)
187       case l: java.util.List[Object@unchecked] =>
188         l.asScala.flatMap(e => makeField(name, e))
189       case prc: ParserRuleContext =>
190         Seq(treeToContainerField(prc))
191       case null =>
192         Seq()
193     }
194     r
195   }
