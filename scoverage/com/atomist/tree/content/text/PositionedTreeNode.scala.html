6 /**
7   * These are the beginnings of TreeNodes. They come in from parsing
8   * They can't be updated, and shouldn't really know their value.
9   *
10   * Goal: these should not implement TreeNode. They aren't path-expression-ready
11   * they have to go through TextTreeNodeLifecycle first, to be turned into
12   * proper TextTreeNodes that know value and can be updated. (Those maintain
13   * position using padding instead of offsets, so the updates work.)
14   *
15   * ParsedNode is the minimal interface that Rug Language Extensions must supply when
16   * parsing a file. It's implemented here for backwards compatibility, so that
17   * existing extensions which supply full PositionedTreeNodes can continue to do so.
18   */
19 trait PositionedTreeNode extends TreeNode with Positioned with ParsedNode {
20 
21   override def parsedNodes: Seq[ParsedNode] = childNodes.map(_.asInstanceOf[ParsedNode])
22 
23   override def startOffset: Int = startPosition.offset
24 
25   override def endOffset: Int = endPosition.offset
26 
27   def hasSamePositionAs(that: PositionedTreeNode): Boolean =
28     this.startPosition.offset == that.startPosition.offset
29 
30   def includes(pos: InputPosition): Boolean =
31     this.startPosition.offset <= pos.offset && this.endPosition.offset >= pos.offset
32 }
33 
34 object PositionedTreeNode {
35   def fromParsedNode(ptn: ParsedNode): PositionedTreeNode = ptn match {
36     case alreadyThere: PositionedTreeNode => alreadyThere
37     case _ =>
38       ImmutablePositionedTreeNode.apply(ptn.nodeName,
39         OffsetInputPosition(ptn.startOffset),
40         OffsetInputPosition(ptn.endOffset),
41         ptn.parsedNodes.map(fromParsedNode),
42         Set(),
43         TreeNode.Signal)
44   }
45 }
