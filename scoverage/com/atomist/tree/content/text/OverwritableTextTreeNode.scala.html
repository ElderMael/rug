250   private def determineLocationStep(visibleChildren: Seq[TreeNode], forChild: TreeNode): String = {
251     val thisChildsName = forChild.nodeName
252     if (!visibleChildren.contains(forChild))
253       throw new IllegalStateException(s"I can't find my child $forChild")
254     else {
255       val childrenBeforeThisWithTheSameName = visibleChildren.takeWhile(_ != forChild).filter(_.nodeName == thisChildsName)
256       val thisChildsIndex = childrenBeforeThisWithTheSameName.size
257       // XPath indexes from 1
258       s"$thisChildsName[${thisChildsIndex + 1}]"
259     }
