9   /**
10    * TreeNodePrinter utility.
11    * It draws a cute tree like
12    *
13 
14 Grandma
15 ├─┬ Dad
16 | ├── Sister
17 | └─┬ Me
18 |   ├── Daughter
19 |   └── Son
20 └── Aunt
21 
22    *
23    * To use it, pass two functions:
24    * @param children how to access child nodes
25    * @param info how to print a node
26    *
27    * @param topOfTree and then the top-level node!
28    *
29    * @return String. Print it yourself
30    */
31   def draw[T](children: T => Seq[T], info: T => String)(topOfTree: T): String = {
32 
33     def drawInternal(tn: T): Seq[String] = {
34       tn match {
35         case leaf if children(leaf).isEmpty => Seq(info(leaf))
36         case parent => info(parent) +: printChildren(children(parent))
37       }
38     }
39 
40     def printChildren(ch: Seq[T]): Seq[String] = {
41       val last = ch.last
42       val notLast = ch.slice(0, ch.length - 1)
43 
44       val lastLine = prefixChildLines(children(last).nonEmpty, last = true, drawInternal(last))
45       val earlierLines = notLast.flatMap(tn => prefixChildLines(children(tn).nonEmpty, last = false, drawInternal(tn)))
46 
47       earlierLines ++ lastLine
48     }
49 
50     drawInternal(topOfTree).mkString("\n")
51   }
52 
53   private def prefixChildLines(hasChildren: Boolean, last: Boolean, childLines: Seq[String]): Seq[String] =
54     childLines.toList match {
55       case head :: rest =>
56         val connector = if (last) FILLER else TREE_CONNECTOR
57         val firstLine = firstLineOfChildPrefix(hasChildren, last) + head
58         val restLines = rest.map(connector + _)
59         firstLine +: restLines
60       case _ => throw new RuntimeException("empty list")
61     }
62 
63   private def firstLineOfChildPrefix(hasChildren: Boolean, last: Boolean): String =
64     if (hasChildren && last) LAST_TREE_NODE_WITH_CHILDREN
65     else if (hasChildren && !last) TREE_NODE_WITH_CHILDREN
66     else if (!hasChildren && last) LAST_TREE_NODE
67     else if (!hasChildren && !last) TREE_NODE
68     else "the impossible has happened"
69
