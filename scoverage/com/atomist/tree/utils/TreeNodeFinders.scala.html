23   }

25   def requiredSingleFieldValue(parent: TreeNode, name: String): String = {
26     parent.childrenNamed(name) match {
27       case Seq(tn) => tn.value
28       case s if s.isEmpty => throw new IllegalArgumentException(s"Found no fields for [$name] in $parent: Needed exactly 1")
29       case more => throw new IllegalArgumentException(s"Found ${more.size} fields for [$name] in $parent: Needed exactly 1")
30     }
31   }
