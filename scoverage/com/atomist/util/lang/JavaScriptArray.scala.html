394 class ReducerJSObject[Y](lst: util.List[Y]) extends AbstractJSObject {
395 
396   override def call(thiz: scala.Any, args: AnyRef*): AnyRef = {
397 
398     // as per https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Array/Reduce
399 
400     if (lst.size() == 0 && args.length < 2) {
401       throw new RuntimeException("TypeError: empty array and no initialValue")
402     }
403 
404     if (lst.size() == 1 && args.length < 2) {
405       return lst.get(0).asInstanceOf[AnyRef]
406     }
407     if (lst.size() == 0 && args.length > 1) {
408       return args(1)
409     }
410 
411     val reduceFn = args.head.asInstanceOf[ScriptObjectMirror]
412     var acc = if (args.length > 1) args(1) else lst.get(0)
413     val idx = if (args.length > 1) 0 else 1
414 
415     for (i <- idx until lst.size()) {
416       acc = reduceFn.call(thiz, acc.asInstanceOf[AnyRef], lst.get(i).asInstanceOf[AnyRef], i.asInstanceOf[AnyRef], lst)
417     }
418     acc.asInstanceOf[AnyRef]
419   }
