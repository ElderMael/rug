13   /**

15     *
16     * @param name property named to convert, which may include hyphen or.
17     * @return Java class name

20     require(name != null && name.nonEmpty, s"Java class name must not be empty or null: [$name] was invalid")

25   private def toJavaIdentifierPart(c: Char): Char =

27     else '_'

29   /**
30     * Convert a string such as big_thing to a Java variable/field/method name like bigThing.
31     * Will guarantee that resulting name is not a reserved word and is a valid Java identifier.
32     *
33     * @param name name, which may include hyphens or underscore. Useful in JSON binding
34     * @return Java variable/field/method name
35     */
36   def toJavaVariableName(name: String): String = {
37     require(name != null && name.nonEmpty, "Property name must not be empty or null")
38     val cleanName = name.toList.map(c => toJavaIdentifierPart(c)).mkString
39     val varName = toCamelizedPropertyName(cleanName)
40     if (JavaParserUtils.isReservedWord(varName) || !Character.isJavaIdentifierStart(varName.charAt(0)))
41       "m_" + varName
42     else varName
43   }
44 
45   /**
46     * Convert a string such as big_thing to a Java property name like bigThing. 
47     *
48     * @param name name, which may include hyphens or underscore. Useful in JSON binding
49     * @return Java property name
50     */
51   def toCamelizedPropertyName(name: String): String = {
52     require(name != null && name.nonEmpty, "Property name must not be empty or null")
53     val tokes = name.split("[-\\._/]")
54     require(tokes.nonEmpty, "Property name must not be empty or null")
55     lowerize(tokes.head) + tokes.tail.map(toke => upperize(toke)).mkString
56   }
57 
58   /**
59     * Take a Java property name and produce a delimited String of all lower case.
60     *
61     * @param javaPropertyName Java property name like bigThing
62     * @return delimited string like big_thing (delimiter is _ in this example)
63     */
64   def toLowerCaseDelimited(javaPropertyName: String, delim: String = "_"): String = {
65     val chars: Seq[String] = javaPropertyName.map(c =>
66       if (c.isUpper) delim + c.toLower else "" + c
67     )
68     chars.mkString("")
69   }
