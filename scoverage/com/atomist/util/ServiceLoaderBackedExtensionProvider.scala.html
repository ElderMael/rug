11 /**
12   * Provider of instances of type [[T]]. Instances are loaded via JDK [[ServiceLoader]] and cached; keyed by their
13   * respective [[ClassLoader]]s.
14   */
15 class ServiceLoaderBackedExtensionProvider[T: ClassTag](val keyProvider: T => String)
16   extends LazyLogging {
17 
18   // The following can be cached as it creates issues in shared class loader hiarchies
19   def providerMap: Map[String, T] = {
20     logger.debug(s"Loading providers of type ${classTag[T].runtimeClass.getName} and class loader ${Thread.currentThread().getContextClassLoader}")
21     ServiceLoader.load(classTag[T].runtimeClass).asScala.map {
22       case t: T =>
23         val key = keyProvider.apply(t)
24         logger.debug(s"Registered provider '$key' with class '${t.getClass}'")
25         key -> t
26       case wtf =>
27         throw new RugRuntimeException("Extension", s"Provider class ${wtf.getClass} must implement ${classTag[T].runtimeClass.getName} interface", null)
28     }.toMap
29   }
30 }
