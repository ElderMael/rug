45   protected def identifierRef(reservedWords: Set[String], underlying: Parser[String] = ident) = new Parser[IdentifierRef] {
46     def apply(in: Input): ParseResult[IdentifierRef] = {
47       val pr = underlying.apply(in)
48       pr match {
49         case succ: Success[String @unchecked] =>
50           if (reservedWords.contains(succ.get))
51             Failure(s"Cannot use reserved word '${succ.get}' as function name", succ.next)
52           else
53             Success[IdentifierRef](IdentifierRef(succ.get), succ.next)
54         case f: Failure => f

57     }
