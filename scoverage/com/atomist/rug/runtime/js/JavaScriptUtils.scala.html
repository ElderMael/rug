135   private def parameterVarToParameter(rug: ScriptObjectMirror, details: ScriptObjectMirror): Parameter = {
136 
137     val pName = details.get("name").asInstanceOf[String]
138     val pPattern = details.get("pattern").asInstanceOf[String]
139     val parameter = Parameter(pName, pPattern)
140     parameter.setDisplayName(details.get("displayName").asInstanceOf[String])
141 
142     details.get("maxLength") match {
143       case x: AnyRef => parameter.setMaxLength(x.asInstanceOf[Int])
144       case _ => parameter.setMaxLength(-1)
145     }
146     details.get("minLength") match {
147       case x: AnyRef => parameter.setMinLength(x.asInstanceOf[Int])
148       case _ => parameter.setMinLength(-1)
149     }
150 
151     parameter.setDefaultRef(details.get("defaultRef").asInstanceOf[String])
152     val disp = details.get("displayable")
153     parameter.setDisplayable(if (disp != null) disp.asInstanceOf[Boolean] else true)
154 
155     if (details.hasMember("required")) {
156       parameter.setRequired(details.get("required").asInstanceOf[Boolean])
157     } else {
158       parameter.setRequired(true)
159     }
160 
161     parameter.addTags(tags(details))
162 
163     parameter.setValidInputDescription(details.get("validInput").asInstanceOf[String])
164     parameter.describedAs(details.get("description").asInstanceOf[String])
165 
166     pPattern match {
167       case s: String if s.startsWith("@") => DefaultIdentifierResolver.resolve(s.substring(1)) match {
168         case Left(_) =>
169           throw new InvalidRugParameterPatternException(s"Unable to recognize predefined validation pattern for parameter $pName: $s")
170         case Right(pat) => parameter.setPattern(pat)
171       }
172       case s: String if !s.startsWith("^") || !s.endsWith("$") =>
173         throw new InvalidRugParameterPatternException(s"Parameter $pName validation pattern must contain anchors: $s")
174       case s: String => parameter.setPattern(s)
175       case _ => throw new InvalidRugParameterPatternException(s"Parameter $pName has no valid validation pattern")
176     }
177 
178     if (rug.hasMember(pName) && rug.getMember(pName) != null) {
179       val asString = rug.getMember(pName).toString
180       if (!parameter.isValidValue(asString))
181         if (asString == "") {
182            // how can we warn them??? this is deprecated
183           println(s"WARNING: On rug ${name(rug)}, parameter '$pName' has a default value of '$asString', but that is not a valid value for this parameter. Set the parameter to null instead, please.")
184         } else {
185           throw new InvalidRugParameterDefaultValue(s"Parameter $pName default value ($asString) is not valid: $parameter")
186         }
187       parameter.setDefaultValue(rug.getMember(pName).toString)
188     }
189 
190     parameter
191   }
