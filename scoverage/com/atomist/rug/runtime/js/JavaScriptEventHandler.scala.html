67   override def handle(ctx: RugContext, e: SystemEvent): Option[Plan] = {
68     val targetNode = ctx.treeMaterializer.rootNodeFor(e, pathExpression)
69     // Put a new artificial root above to make expression work
70     val root = JavaScriptEventHandler.rootNodeFor(targetNode)
71     ctx.pathExpressionEngine.ee.evaluate(root, pathExpression, ctx, None) match {
72       case Right(Nil) => None
73       case Right(matches) if matches.nonEmpty =>
74         val cm = jsContextMatch(
75           wrapOne(targetNode, ctx.typeRegistry),
76           wrap(matches, ctx.typeRegistry),
77           ctx.pathExpressionEngine,
78           ctx.contextRoot(),
79           teamId = e.teamId)
80         invokeMemberFunction(
81           jsc,
82           handler,
83           "handle",
84           None,
85           jsScalaHidingProxy(cm, returnNotToProxy = _ => true)
86         ) match {
87           case plan: ScriptObjectMirror => ConstructPlan(plan, Some(this))
88           case other => throw new InvalidHandlerResultException(s"$name EventHandler returned an invalid response ($other) when handling $pathExpressionStr")
89         }
90       case Right(matches) if matches.isEmpty => None
91       case Left(failure) =>
92         throw new RugRuntimeException(pathExpressionStr,
93           s"Error evaluating path expression $pathExpression: [$failure]")
94     }
95   }
