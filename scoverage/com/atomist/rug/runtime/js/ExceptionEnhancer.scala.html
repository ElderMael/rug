24 /**
25   * Enhances exceptions caught within a JavaScript context that
26   * we have all sources for
27   */
28 object ExceptionEnhancer {
29 
30   def enhanceIfPossible(rugAs: ArtifactSource, ecmaEx: ECMAException): Exception = {
31 
32     // Let this through. They probably threw it voluntarily.
33     if (ecmaEx.thrown.isInstanceOf[NativeError])
34       throw ecmaEx
35 
36     ecmaEx.getFileName match {
37       case "<eval>" | null =>
38         // Can't add much useful info
39         ecmaEx
40       case filePath =>
41         val f = rugAs.findFile(filePath).getOrElse(
42           throw new IllegalStateException(s"Cannot find file at path [$filePath] attempting to handle $ecmaEx")
43         )
44 
45         val message: String =
46           Objects.toString(ecmaEx.getThrown match {
47           case nte: NativeTypeError =>
48             nte.instMessage
49           case x => x
50         })
51 
52         val line = ecmaEx.getLineNumber
53         val col = ecmaEx.getColumnNumber.max(1)
54         // Nashorn can report -1, which breaks source maps
55         val pos = LineInputPositionImpl(f.content, line, col)
56         val jsri = RuntimeErrorInfo(message,
57           f.path, pos, ecmaEx.getThrown)
58         findSourceMappedSourceInformation(rugAs, jsri) match {
59           case Some(rti) => new SourceLanguageRuntimeException(jsri, rti)
60           case None => new JavaScriptRuntimeException(jsri)
61         }
62     }
63   }
64 
65   private def findSourceMappedSourceInformation(rugAs: ArtifactSource, jsri: RuntimeErrorInfo): Option[RuntimeErrorInfo] = {
66     val sourceMapName = jsri.filePath + ".map"
67     rugAs.findFile(sourceMapName).flatMap(map => {
68       val sourceMap = new SourceMapImpl(map.content)
69       val mapping = sourceMap.getMapping(jsri.pos.lineFrom1, jsri.pos.colFrom1)
70       Option(mapping).flatMap(mapping => {
71         val sourcePath = jsri.filePath.substring(0, jsri.filePath.lastIndexOf("/"))
72         val sourceFileO = rugAs.findFile(s"$sourcePath/${mapping.getSourceFileName}")
73         sourceFileO.map(sourceFile => {
74           val lip = LineInputPositionImpl(sourceFile.content,
75             mapping.getSourceLine + 1,
76             mapping.getSourceLine)
77           RuntimeErrorInfo(jsri.message, sourceFile.path, lip, jsri.detail)
78         })
79       })
80     })
81   }
82 
83 }
