18 class JavaScriptCommandHandlerFinder

22   /**

24     */
25   def isValid(obj: ScriptObjectMirror): Boolean = {
26     obj.getMember("__kind") == "command-handler" &&

29   }

31   override def create(jsc: JavaScriptContext,

41       tags(handler),
42       secrets(handler),

44       testDescriptor(handler)))
45   }
46 
47   /**
48     * Extract any test related metadata - if any.
49     */
50   protected def testDescriptor(someVar: ScriptObjectMirror): Option[TestDescriptor] = {
51     someVar.getMember("__test") match {
52       case som: ScriptObjectMirror => (som.getMember("description"), som.getMember("kind")) match {
53         case (description: String, kind: String) => Some(TestDescriptor(kind, description))
54         case _ => throw new InvalidTestDescriptorException("A test must have a 'kind' and a 'description'")
55       }
56       case _ => None
57     }
58   }
59   /**
60     * Extract intent from a var.
61     */
62   protected def intent(someVar: ScriptObjectMirror): Seq[String] = {
63     someVar.getMember("__intent") match {
64       case som: ScriptObjectMirror =>
65         val stringValues = som.values().asScala collect {
66           case s: String => s
67         }
68         stringValues.toSeq
69       case _ => Nil
70     }
71   }
72 
73   /**
74     * Look for mapped parameters (i.e. defined by the bot).
75     *
76     * See MappedParameters in Handlers.ts for examples.
77     */
78   protected def mappedParameters(someVar: ScriptObjectMirror): Seq[MappedParameter] = {
79     getMember(someVar, "__mappedParameters") match {
80       case Some(ps: ScriptObjectMirror) if !ps.isEmpty =>
81         ps.asScala.collect {
82           case (_, details: ScriptObjectMirror) =>
83             val localKey = details.getMember("localKey").asInstanceOf[String]
84             val foreignKey = details.getMember("foreignKey").asInstanceOf[String]
85             MappedParameter(localKey, foreignKey)
86         }.toSeq
87       case _ => Seq()
88     }
89   }
90 }

111     */
112   override def handle(ctx: RugContext, params: ParameterValues): Option[Plan] = {
113     val validated = addDefaultParameterValues(params)
114     validateParameters(validated)
115     // We need to proxy the context to allow for property access
116     invokeMemberFunction(
117       jsc,
118       handler,
119       "handle",
120       Some(validated),
121       jsScalaHidingProxy(ctx, returnNotToProxy = jsSafeCommittingProxy.DoNotProxy)) match {
122       case plan: ScriptObjectMirror =>
123         ConstructPlan(plan, Some(this))
124       case other =>
125         throw new InvalidHandlerResultException(s"$name CommandHandler did not return a recognized response ($other) when invoked with ${params.toString()}")
126     }
127   }
