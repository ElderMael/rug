36     handler.getMember("__coercion") match {

63   override def handle(ctx: RugContext, response: Response, params: ParameterValues): Option[Plan] = {
64     // TODO this handle method is almost identical to the command handler - extract it
65     val validated = addDefaultParameterValues(params)
66     validateParameters(validated)
67     val coerced = responseCoercer.coerce(response)
68     invokeMemberFunction(
69       jsc,
70       handler,
71       "handle",
72       Some(validated),
73       jsScalaHidingProxy(jsResponse(coerced.msg.orNull, coerced.code.getOrElse(-1), coerced.body.getOrElse(Nil))),
74       jsScalaHidingProxy(ctx)) match {
75       case plan: ScriptObjectMirror => ConstructPlan(plan, Some(this))
76       case other => throw new InvalidHandlerResultException(s"$name ResponseHandler did not return a recognized response ($other) when invoked with ${params.toString()}")
77     }
78   }

80   /**

82     * but only if no @Parameter annotations are there
83     * This is handy to avoid use @Parameter decorators
84     */
85   override protected def setParameters(clone: ScriptObjectMirror, params: Seq[ParameterValue]): Unit = {
86     super.setParameters(clone, params)

92       })
93     }
