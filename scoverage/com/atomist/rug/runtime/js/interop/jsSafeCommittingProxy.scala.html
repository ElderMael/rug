85   override def getMember(name: String): AnyRef = {
86     // First, look for a member added in JavaScript
87     val member = additionalMembers.get(name) match {
88       case Some(som: ScriptObjectMirror) =>
89         som
90       case Some(x) =>
91         x
92       case None if MagicJavaScriptMethods.contains(name) =>
93         super.getMember(name)
94       case None if name == "toString" =>
95         new AlwaysReturns(toString)
96       case None if name == "$jumpInto" =>
97         new SquirrelFunction(array = true)
98       case None if name == "$jumpIntoOne" =>
99         new SquirrelFunction(array = false)
100       case _ =>
101         // Invoke using navigation on underlying node or reflection
102         invokeConsideringTypeInformation(name)
103     }
104     logger.debug(s"Returning member [$member] for $name in $this")
105     member

140   private def invokeConsideringTypeInformation(name: String): AnyRef = {
141     val possibleOps = typ.allOperations.filter(op => name == op.name)
142     if (possibleOps.nonEmpty) {
143       val op = possibleOps.head //TODO seems fragile
144       if (op.invocable) {
145         val function = new FunctionProxyToReflectiveInvocationOnUnderlyingJVMNode(name, possibleOps)
146         if (op.exposeAsProperty) {
147           // Reuse the logic we have in the function implementation by simply creating and invoking it
148           function.call("whatever")
149         }
150         else {
151           function
152         }
153       }
154       else
155         nodeNavigationPropertyAccess(node, name)
156     }
157     else
158       invokeGivenNoMatchingOperationInTypeInformation(name, typ)

161   private def invokeGivenNoMatchingOperationInTypeInformation(name: String, st: Typed): AnyRef = {
162     if (node.hasTag(TreeNode.Dynamic)) name match {
163       case navigation if node.relatedNodeNames.contains(navigation) =>

170         // This object is wholly defined in JavaScript

181   /**
182     * Nashorn proxy for a method invocation that delegates to the
183     * underlying node using reflection.
184     */
185   private class FunctionProxyToReflectiveInvocationOnUnderlyingJVMNode(name: String, possibleOps: Seq[TypeOperation])
186     extends AbstractJSObject {
187 
188     override def isFunction: Boolean = true
189 
190     override def call(thiz: scala.Any, args: AnyRef*): AnyRef = {
191       possibleOps.find(op => op.parameters.size == args.size) match {
192         case None =>
193           logger.warn(
194             s"Attempt to invoke method [$name] on type [${typ.description}] with ${args.size} arguments: No matching signature")
195           UNDEFINED
196         case Some(op) =>
197           // Reflective invocation
198           val returned = op.invoke(node, args.toSeq)
199           node match {
200             case c: MutableView[_] if !op.readOnly => c.commit()
201             case _ =>
202           }
203           // The returned type needs to be wrapped if it's a collection
204           wrapJVMReturn(returned, op)
205       }
206     }
207 
208     import scala.collection.JavaConverters._
209 
210     private def wrapJVMReturn(returned: AnyRef, op: TypeOperation): AnyRef = {
211       returned match {
212         case l: util.List[_] => new JavaScriptArray(wrapIfNecessary(l.asScala, typeRegistry))
213         case s: Seq[_] => new JavaScriptArray(wrapIfNecessary(s, typeRegistry))
214         case s: Set[_] => new JavaScriptArray(s.toSeq.asJava)
215         case r =>
216           // Be sure to proxy all the way down
217           val wrapped = wrapIfNecessary(Seq(r), typeRegistry).get(0)
218           if (op.exposeResultDirectlyToNashorn)
219             jsScalaHidingProxy(wrapped, returnNotToProxy = DoNotProxy)
220           else
221             wrapped
222       }
223     }
224   }

227     * Nashorn property access using navigation methods on TreeNode.
228     */
229   private def nodeNavigationPropertyAccess(node: GraphNode, name: String): AnyRef = {
230 
231     import scala.language.reflectiveCalls
232     val nodesAccessedThroughThisFunctionCall: Seq[GraphNode] = node.relatedNodesNamed(name)
233     nodesAccessedThroughThisFunctionCall.toList match {
234       case Nil =>
235         logger.warn(name,
236           s"No children or function found for property '$name' on $node")
237         UNDEFINED
238       case null :: Nil => null
239       case (ttn: TerminalTreeNode) :: Nil if ttn.hasTag(Cardinality.One2Many) =>
240         // Pull out the value and put in an array
241         JavaScriptArray.fromSeq(Seq(ttn.value))
242       case (ttn: TerminalTreeNode) :: Nil =>
243         // Pull out the value
244         ttn.value
245       case head :: Nil if !head.hasTag(Cardinality.One2Many) =>
246         // Only one entry and we know it's not marked as an array
247         wrapOne(head)
248       case more => new JavaScriptArray(wrapIfNecessary(more, typeRegistry))
249     }
250   }
