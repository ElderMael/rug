14 /**
15   * Use this if you want to expose a structure in JavaScript
16   * with TypeScript-friendly arrays and null in place of Option.
17   * Simply uses reflection.
18   * Also, enables suppression of some methods.
19   * Exposes no-arg methods as properties but allows invocation with parameters.
20   * If you wish to expose a no-arg method as a function (because it has
21   * side effects etc) use
22   * @see ExposeAsFunction
23   */

31   override def getMember(name: String): AnyRef = {
32     val r = name match {
33       case _ if methodsToHide.contains(name) =>
34         // Make JavaScript folks feel at home.
35         ScriptRuntime.UNDEFINED
36       case _ if jsSafeCommittingProxy.MagicJavaScriptMethods.contains(name) =>
37         super.getMember(name)
38       case _ =>
39         try {
40           ReflectionUtils.getAllDeclaredMethods(target.getClass).find(_.getName == name) match {
