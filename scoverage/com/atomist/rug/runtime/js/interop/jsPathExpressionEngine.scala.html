77   /**
78     * Evaluate the given path expression.

95   private def evaluateParsed(rootAsObject: AnyRef, parsed: PathExpression) = {
96     val root = toUnderlyingGraphNode(rootAsObject)
97     val hydrated = rugContext.treeMaterializer.hydrate(rugContext.teamId, toUnderlyingGraphNode(root), parsed)
98     ee.evaluate(hydrated, parsed,
99       SimpleExecutionContext(typeRegistry, rugContext.repoResolver)
100     ) match {
101       case Right(nodes) =>
102         val m = Match(root, wrap(nodes, typeRegistry))
103         m
104       case Left(_) =>
105         Match(root, Collections.emptyList())
106     }
107   }

109   private def toUnderlyingGraphNode(o: AnyRef): GraphNode = o match {

121   /**
122     * Evaluate the path expression, applying a function
123     * to each result.
124     *
125     * @param root  node to evaluate path expression against
126     * @param pexpr path expression (compiled or string)
127     * @param f     function to apply to each path expression
128     */
129   def `with`(root: AnyRef, pexpr: Object, f: Object): Unit = f match {
130     case som: ScriptObjectMirror =>
131       val r = evaluateInternal(root, pexpr)
132       r.matches.asScala.foreach(m => {
133         val args = Seq(m)
134         // println(s"I am calling ${som} with value ${m}")
135         som.call("apply", args: _*)
136       })
137     case _ =>
138       throw new RugRuntimeException(null, s"Invalid argument to 'with' method: $f is not a JavaScript function")
139   }

141   /**
142     * Return a single match to the given path expression. Throw an exception otherwise.
143     *
144     * @param root root of Tree. SafeCommittingProxy wrapping a TreeNode
145     * @param pe   path expression of object
146     */
147   def scalar(root: AnyRef, pe: Object): GraphNode = {
148     val res = evaluateInternal(root, pe)
149     val ms = res.matches
150     ms.size() match {
151       // TODO use more specific exception type
152       case 0 => throw new Exception(s"No matches found for path expression [${jsPathExpressionEngine.pathExpressionFromObject(pe)}]")
153       case 1 => ms.get(0)
154       case more => throw new PathExpressionException(root, pe, s"Too many matches found for path expression [${jsPathExpressionEngine.pathExpressionFromObject(pe)}! Found $more")
155     }
156   }

158   private def matchReport(root: AnyRef, pe: Object) = {
159     val pathExpression = jsPathExpressionEngine.pathExpressionFromObject(pe)
160 
161     def inner(report: Seq[String], lastEmptySteps: Option[PathExpression], steps: PathExpression): Seq[String] = {
162       if (steps.locationSteps.isEmpty) {
163         report // nowhere else to go
164       } else evaluateParsed(root, steps).matches match {
165         case empty if empty.isEmpty => // nothing found, keep looking
166           inner(report, Some(steps), steps.dropLastStep)
167         case nonEmpty => // something was found
168           val dirtyDeets = if (nonEmpty.size > 1) "" else s" = ${nonEmpty.get(0)}"
169           val myReport = Seq(s"$steps found ${nonEmpty.size}$dirtyDeets")
170           val recentEmptyReport = lastEmptySteps.map(" " + _ + " found 0").toSeq
171           val reportSoFar = myReport ++ recentEmptyReport ++ report
172           inner(reportSoFar.map(" " + _), None, steps.dropLastStep)
173       }
174     }
175 
176     inner(Seq(), None, pathExpression).mkString("\n")
177   }
