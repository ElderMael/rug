18   /**
19     * Get all key values, ignoring exceptions, which can be thrown
20     * by Nashorn entrySet
21     */
22   private def safeObjectMap(som: ScriptObjectMirror): Map[String, AnyRef] =
23     som.keySet().asScala
24       .flatMap(key => {
25         // TypeScript "get" properties can throw exceptions
26         val maybeValue = allCatch.opt(som.get(key))
27         maybeValue.map(value => key -> value)
28       }).toMap
29

31   def extractProperties(som: ScriptObjectMirror): Map[String, AnyRef] =
32     safeObjectMap(som) flatMap {
33       case (_, som: ScriptObjectMirror) if som.isFunction =>
34         None
35       case (key, value) =>
36         Some(key -> value)
37     }
38 
39   /**
40     * Return the current state of no-arg methods on this object
41     */
42   def extractNoArgFunctionValues(som: ScriptObjectMirror): Map[String, AnyRef] = {
43     val m = safeObjectMap(som) flatMap {
44       case (key, f: ScriptObjectMirror) if isNoArgFunction(f) =>
45         // If calling the function throws an exception, discard the value.
46         // This will happen with builder stubs that haven't been fully initialized
47         // Otherwise, use it
48         allCatch.opt(som.callMember(key))
49           .map(result => {
50             (key, result)
51           })
52       case _ => None
53     }
54     m
55   }

65   def toJavaType(nashornReturn: Object): Object = nashornReturn match {

72   def toJavaMap(nashornReturn: Object): Map[String, Object] =
73     nashornReturn match {
74       case som: ScriptObjectMirror =>
75         val scalaMap = som.entrySet().asScala.map {
76           e: Entry[String, Object] => (e.getKey, e.getValue)
77         }.toMap
78         scalaMap.mapValues {
79           case som: ScriptObjectMirror =>
80             toJavaMap(som)
81           case x =>
82             toJavaType(x)
83         }
84     }

102   /**
103     * Call the given JavaScript function, which must return a string
104     */
105   def stringFunction(som: ScriptObjectMirror, name: String): String =
106     som.callMember(name) match {
