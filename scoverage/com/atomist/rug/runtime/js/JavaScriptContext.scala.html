83   /**
84     * Evaluate the given file, with proper exports handling.
85     * NB: Must be a file from atomistContent.
86     */
87   def evaluate(f: FileArtifact): Unit = {
88     val varName = f.path.dropRight(3).replaceAll("/", "_").replaceAll("\\.", "\\$")
89     try {
90       // Because otherwise the loader doesn't know about the paths and can't resolve relative modules
91       val toEval = s"exports.$varName = require('./${f.path.dropRight(3)}');"
92       // TODO is it safe to keep calling this?
93       engine.put(ScriptEngine.FILENAME, f.path)
94       engine.eval(toEval)
95     }
96     catch {
97       case x: ScriptException => throw new RugJavaScriptException(s"Error during eval of: ${f.path}", x)
98       case x: RuntimeException => x.getCause match {
99         case c: ScriptException => throw new RugJavaScriptException(s"Error during eval of: ${f.path}", c)
100         case _ => throw x
101       }
102     }
103   }

119   /**
120     * Return all the vars known to the engine that expose ScriptObjectMirror objects, with the key.
121     *
122     * @return ScriptObjectMirror objects for all vars known to the engine

124   def vars: Seq[Var] = {
125     engine.getContext.getBindings(ScriptContext.ENGINE_SCOPE)
126       .get("exports").asInstanceOf[ScriptObjectMirror]
127       .asScala.collect {
128       case (_, v: ScriptObjectMirror) => extractVars(v)

132   private def extractVars(obj: ScriptObjectMirror): Seq[Var] = {
133     obj.entrySet().asScala.flatMap(e => {
134       (e.getKey, e.getValue) match {
135         case (k, som: ScriptObjectMirror) => Some(Var(k, som))
136         case _ => None
137       }
138     }).toSeq
139   }

144     private val commentPattern: Pattern = Pattern.compile("^//.*$", Pattern.MULTILINE)

146     def this(artifacts: ArtifactSource, rootPath: String = "") {

150     def getFile(s: String): String = {
151       val file = artifacts.findFile(getPath + s)
152       if (file.isEmpty) return null
153       //remove these source-map comments because they seem to be breaking nashorn :/
154       commentPattern.matcher(file.get.content).replaceAll("")
155     }
156 
157     def getFolder(s: String): Folder = {
158       val dir = artifacts.findDirectory(getPath + s)
159       if (dir.isEmpty) return null
160       new ArtifactSourceBasedFolder(artifacts, this, getPath + s + "/")
161     }
162   }
