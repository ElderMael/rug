29   def constructPlan(jsPlan: ScriptObjectMirror, returningRug: Option[Rug]): Plan = {
30 
31     val jsMessages = jsPlan.getMember("messages") match {
32       case o: ScriptObjectMirror => o.values().toArray.toList
33       case _ => Nil
34     }
35     val messages: Seq[Message] = jsMessages.map { message =>
36       val m = message.asInstanceOf[ScriptObjectMirror]
37       constructMessage(m)
38     }
39     val instructions = jsPlan.getMember("instructions") match {
40       case u: Undefined => Nil
41       case jsInstructions: ScriptObjectMirror =>
42         jsInstructions.values().toArray.toList.map { respondable =>
43           constructRespondable(respondable.asInstanceOf[ScriptObjectMirror], returningRug)
44         }
45     }
46 
47     val lifecycle = messages.collect { case o: LifecycleMessage => o }
48     val local = messages.collect { case o: LocallyRenderedMessage => o }
49 
50     Plan(returningRug, lifecycle, local, instructions, nativeObject = Some(jsPlan))
51   }

53   def constructMessage(jsMessage: ScriptObjectMirror): Message = {
54 
55     jsMessage.getMember("kind") match {
56       case "response" | "directed" =>
57         val messageBody = jsMessage.getMember("body") match {
58           case json: ScriptObjectMirror =>
59             throw new UnsupportedOperationException("Message body must be a string")
60           case text: String => text
61           case _ =>
62             throw new InvalidHandlerResultException(s"Cannot determine message content from body: ${jsMessage.getMember("body")}")
63         }
64         val contentType = jsMessage.getMember("contentType") match {
65           case c: String => c
66           case _ => throw new InvalidHandlerResultException(s"Message must have a `contentType`")
67         }
68         val usernames = jsMessage.getMember("usernames") match {
69           case _: Undefined => Nil
70           case usernames: ScriptObjectMirror =>
71             usernames.values().toArray.toSeq.asInstanceOf[Seq[String]]
72         }
73         val channelNames = jsMessage.getMember("channelNames") match {
74           case _: Undefined => Nil
75           case channelNames: ScriptObjectMirror =>
76             channelNames.values().toArray.toSeq.asInstanceOf[Seq[String]]
77         }
78         val instructions = jsMessage.getMember("instructions") match {
79           case _: Undefined => Nil
80           case jsInstructions: ScriptObjectMirror =>
81             jsInstructions.values().toArray.toList.map { presentable =>
82               constructPresentable(presentable.asInstanceOf[ScriptObjectMirror])
83             }
84         }
85         LocallyRenderedMessage(messageBody, contentType, channelNames, usernames, instructions)
86 
87       case "lifecycle" =>
88         val instructions = jsMessage.getMember("instructions") match {
89           case _: Undefined => Nil
90           case jsInstructions: ScriptObjectMirror =>
91             jsInstructions.values().toArray.toList.map { presentable =>
92               constructPresentable(presentable.asInstanceOf[ScriptObjectMirror])
93             }
94         }
95 
96         val node = jsMessage.getMember("node") match {
97           case t: GraphNode => t
98           case som: ScriptObjectMirror =>
99             NashornMapBackedGraphNode.toGraphNode(som).getOrElse(
100               throw new InvalidHandlerResultException(s"Lifecycle message node script could not be converted to a GraphNode: Invalid argument: $som")
101             )
102           case a =>
103             throw new InvalidHandlerResultException(s"Lifecycle messages must contain a GraphNode, actually: $a")
104         }
105 
106         val id = jsMessage.getMember("lifecycleId") match {
107           case s: String => s
108           case _ => throw new InvalidHandlerResultException("Lifecycle messages must contain a lifecycleId")
109         }
110         LifecycleMessage(node, instructions, id)
111 
112       case _: Undefined => throw new InvalidHandlerResultException(s"A message must have a kind: $jsMessage")
113     }
114   }

116   def constructPresentable(jsPresentable: ScriptObjectMirror): Presentable = {
117     val instruction = jsPresentable.getMember("instruction") match {
118       case o: ScriptObjectMirror => constructInstruction(o)
119       case x => throw new InvalidHandlerResultException(s"A Message Presentable must contain an 'Instruction', actually $x")

126       case x: String => Some(x)
127       case _ => None
128     }

132   def constructRespondable(jsRespondable: ScriptObjectMirror, returningRug: Option[Rug]): Plannable = {
133     val instruction: Instruction = jsRespondable.getMember("instruction") match {
134       case u: Undefined =>
135         throw new IllegalArgumentException(s"No instruction found in $jsRespondable")

143       case r: RespondableInstruction => Respondable(r, onSuccess, onError)

147   def constructInstruction(jsInstruction: ScriptObjectMirror): Instruction = {
148     val jsInstructionKind = jsInstruction.getMember("kind") match {

156   def constructInstructionDetail(jsInstruction: ScriptObjectMirror): Instruction.Detail = {
157     val parameters: Seq[ParameterValue] = jsInstruction.getMember("parameters") match {
158       case o: ScriptObjectMirror =>
159         val filtered = o.keySet().toArray.filter(key => {
160           val name = key.asInstanceOf[String]
161           o.getMember(name) match {
162             case null => false
163             case ScriptRuntime.UNDEFINED => false
164             case o: ScriptObjectMirror if o.isFunction => false
165             case _ => true
166           }
167         })
168 
169         filtered.map { key =>
170           val name = key.asInstanceOf[String]
171           val value = o.getMember(name)
172           SimpleParameterValue(name,
173             value match {
174               case s: String => s
175               case o => JsonUtils.toJson(o)
176             })
177         }
178       case _ => Nil
179     }
180 
181     val (name, coordinates) = jsInstruction.getMember("name") match {
182       case name: String =>
183         (name, None)
184       case o: ScriptObjectMirror =>
185         val name = o.getMember("name").asInstanceOf[String]
186         val coordinates = MavenCoordinate(
187           o.getMember("group").asInstanceOf[String],
188           o.getMember("artifact").asInstanceOf[String])
189         (name, Some(coordinates))
190       case x => throw new InvalidHandlerResultException(s"Instruction name cannot be $x")
191     }
192 
193     val project_name = jsInstruction.getMember("project") match {
194       case project_name: String => Some(project_name)
195       case _ => None
196     }
197 
198     val commitMessage = jsInstruction.getMember("commitMessage") match {
199       case msg: String => Some(msg)
200       case _ => None
201     }
202 
203     val target = jsInstruction.getMember("target") match {
204       case o: ScriptObjectMirror if o.hasMember("baseBranch") =>
205         val baseBranch = o.getMember("baseBranch").asInstanceOf[String]
206         o.getMember("kind") match {
207           case "github-pull-request" =>
208             val title = if (o.hasMember("title")) Some(o.getMember("title").asInstanceOf[String]) else None
209             val body = if (o.hasMember("body")) Some(o.getMember("body").asInstanceOf[String]) else None
210             val headBranch = if (o.hasMember("headBranch")) Some(o.getMember("headBranch").asInstanceOf[String]) else None
211             Some(GitHubPullRequest(baseBranch, commitMessage, headBranch, title, body))
212           case "github-branch" =>
213             val headBranch = o.getMember("headBranch") match {
214               case b: String => Some(b)
215               case _ => None
216             }
217             Some(GitHubBranch(baseBranch, commitMessage, headBranch))
218           case k => throw new BadPlanException(s"Unsupported EditorTarget kind: $k")
219         }
220       case _ =>
221         Some(DefaultTarget(commitMessage))
222     }
223 
224     Instruction.Detail(name, coordinates, parameters, project_name, target)
225   }

227   def constructCallback(callback: Object, returningRug: Option[Rug]): Option[Callback] = {
228     callback match {
229       case jsOnSuccess: ScriptObjectMirror =>
230         val callback = jsOnSuccess.getMember("kind") match {
231           case "response" | "lifecycle" | "directed" => constructMessage(jsOnSuccess)
232           case _: String => Respond(constructInstructionDetail(jsOnSuccess))
233           case _ =>
234             if (jsOnSuccess.hasMember("messages") || jsOnSuccess.hasMember("instructions")) {
235               constructPlan(jsOnSuccess, returningRug)
236             }
237             else {
238               throw new InvalidHandlerResultException(s"Cannot create CallBack from: $jsOnSuccess")
239             }
240         }
241         Option(callback)
242       case _ => None
243     }
244   }
