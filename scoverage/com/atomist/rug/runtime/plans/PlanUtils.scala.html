47   private def planToTree(plan: Plan): BabyTree = {
48 
49     def respondableToTree(plannable: Plannable): BabyTree = {
50       val i = instructionToString(plannable.instruction)
51       plannable match {
52         case r: Respondable =>
53           val onSuccessChild = r.onSuccess.map(callbackToTree("onSuccess", _))
54           val onFailureChild = r.onFailure.map(callbackToTree("onFailure", _))
55           BabyTree(i, onSuccessChild.toSeq ++ onFailureChild)
56         case nr: Nonrespondable =>
57           BabyTree(i, Nil)
58       }
59     }
60 
61     val allMessages = plan.local ++ plan.lifecycle
62     BabyTree("Plan",
63       plan.instructions.sortBy(_.toString).map(respondableToTree) ++
64         allMessages.sortBy(_.toDisplay).map(messageToTree))
65   }

67   private def awaitAndTreeLogEvents(name: String, events: Seq[PlanLogEvent]): BabyTree = {
68     def logEventToTree(one: PlanLogEvent): BabyTree = {
69       def nullSafeMessage(x: Throwable) = s"Error: ${Option(x).map(_.getMessage).getOrElse("null exception")}"
70 
71       one match {
72         case InstructionError(i, x) => BabyTree(nullSafeMessage(x), Seq(BabyTree(instructionToString(i))))
73         case MessageDeliveryError(m, x) => BabyTree(nullSafeMessage(x), Seq(messageToTree(m)))
74         case CallbackError(c, x) => BabyTree(nullSafeMessage(x), Seq(callbackToTree("failed callback", c)))
75         case InstructionResult(instruction, response) =>
76           BabyTree(s"Received $response back from ${instructionToString(instruction)}")
77         case NestedPlanRun(plan, planResult) =>
78           Await.ready(planResult, 10.seconds)
79           val result = planResult.value.get match {
80             case Failure(exception) => BabyTree(s"Future failed with ${exception.getMessage}")
81             case Success(value) => awaitAndTreeLogEvents(s"Future completed", value.log)
82           }
83           val planChild = planToTree(plan)
84           BabyTree("Nested plan run", Seq(planChild, result))
85       }
86     }
87 
88     val children = events.map(logEventToTree)
89     BabyTree(name, children.sortBy(_.print))
90   }
