35   private def doWithProjectName(instruction: Instruction, action: (String) => Response) = {

42   override def run(instruction: Instruction, callbackInput: Option[Response]): Response = {
43     val parameters = SimpleParameterValues(instruction.detail.parameters)
44     instruction match {
45       case Execute(detail) =>
46         rugFunctionRegistry.find(detail.name) match {
47           case Some(fn) =>
48             val replaced = secretResolver.replaceSecretTokens(detail.parameters)
49             val resolved = SimpleParameterValues(replaced ++ secretResolver.resolveSecrets(fn.secrets))
50             Try {
51               fn.run(resolved)
52             } match {
53               case ScalaSuccess(response) =>
54                 //ensure the body is String or byte[]!
55                 val thedata = response.body match {
56                   case Some(Body(Some(str), None)) => Some(str)
57                   case Some(Body(None, Some(bytes))) => Some(bytes)
58                   case Some(Body(_, _)) => throw new BadRugFunctionResponseException(s"Function `${fn.name}` should return a string body or a byte array, but not both")
59                   case _ => None
60                 }
61                 Response(response.status, response.msg, response.code, thedata)
62               case ScalaFailure(throwaball) =>
63                 val msg = s"Rug Function ${detail.name} threw exception: ${throwaball.getMessage}"
64                 logger.warn(msg, throwaball)
65                 Response(Status.Failure, Some(msg), None, Some(throwaball))
66             }
67           case _ =>
68             val msg = s"Cannot find Rug Function ${detail.name}"
69             logger.warn(msg)
70             Response(Status.Failure, Some(msg), None, None)
71         }
72       case _ =>
73         rugResolver match {
74           case Some(resolver) =>
75             resolver.resolve(currentRug, extractName(instruction.detail)) match {
76               case Some(rug: ProjectGenerator) =>
77                 doWithProjectName(instruction, (projectName: String) => {
78                   val as = projectManagement.generate(rug, parameters, projectName)
79                   if (as != null) Response(Success)
80                   else Response(Failure, Some(s"failed to run generator ${rug.name} to create $projectName"))
81                 })
82               case Some(rug: ProjectEditor) =>
83                 doWithProjectName(instruction, (projectName: String) => {
84                   projectManagement.edit(rug, parameters, projectName, instruction.detail.editorTarget) match {
85                     case _: SuccessfulModification => Response(Success)
86                     case success: NoModificationNeeded => Response(Success, Some(success.comment))
87                     case failure: FailedModificationAttempt => Response(Failure, Some(failure.failureExplanation))
88                   }
89                 })
90               case Some(rug: CommandHandler) =>
91                 val planOption = rug.handle(rugContext, parameters)
92                 Response(Success, None, None, planOption)
93               case Some(rug: ResponseHandler) =>
94                 callbackInput match {
95                   case Some(response) =>
96                     val planOption = rug.handle(rugContext, response, parameters)
97                     Response(Success, None, None, planOption)
98                   case c =>
99                     throw new BadPlanException(s"Callback input was not recognized: $c")
100                 }
101               case Some(rug) => throw new BadPlanException(s"Unrecognized rug type: $rug")
102               case None => throw new BadPlanException(s"Could not find rug with name: ${instruction.detail.name}")
103             }
104           case _ => throw new IllegalArgumentException(s"Could not find rug with name: ${instruction.detail.name} because no RugResolver supplied")
105         }
106     }
107   }
108

113     detail.coordinates match {
114       case Some(coords) =>
115         s"${coords.group}:${coords.artifact}:${detail.name}"
116       case _ => detail.name
117     }
118   }
