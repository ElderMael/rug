15   /**
16     * Replace things of the form #{secret_path} with secret values in supplied ParameterValues
17     *
18     * Any replaced secrets _must_ be declared on the handler itself
19     *
20     * Used to inject secrets in to RugFunction parameters in LocalInstructionRunner.
21     */
22   def replaceSecretTokens(parameters: Seq[ParameterValue]): Seq[ParameterValue] = {
23     rug match {
24       case Some(handler: SecretAwareRug) =>
25         parameters.map { param =>
26           param.getValue match {
27             case s: String =>
28               val replaced = handler.secrets.foldLeft[String](s) {(cur, secret) =>
29                 cur.replace(s"#{${secret.path}}", resolveSecret(secret).getValue.toString)
30               }
31               SimpleParameterValue(param.getName,replaced)
32             case _ => param
33           }
34         }
35       case _ => parameters
36     }
37   }

42     * @param secrets set of Secrets
43     * @return a mapping from secret name (could be same as path) to actual secret value
44     */
