15   def interpret(planResult: PlanResult): Response = {

17       Response(Failure)
18     } else {
19       Response(Success)
20     }
21   }
22 
23   @tailrec
24   def hasLogFailure(log: Seq[PlanLogEvent]): Boolean = {
25     log.headOption match {
26       case Some(head) =>
27         head match {
28           case _: PlanLogError => true
29           case result: InstructionResult if result.response.status == Failure => true
30           case result: NestedPlanRun =>
31             val planResult = Await.result(result.planResult, 5.minutes)
32             hasLogFailure(log.tail ++ planResult.log)
33           case _ => hasLogFailure(log.tail)
34         }
35       case None => false
36     }
37   }
