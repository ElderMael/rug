22   override def run(plan: Plan, callbackInput: Option[Response]): Future[PlanResult] = {
23     val allMessages = plan.lifecycle ++ plan.local
24     val messageLog: Seq[MessageDeliveryError] = allMessages.flatMap { message =>
25       if (plan.returningRug.nonEmpty) {
26         Try(messageDeliverer.deliver(plan.returningRug.get, message, callbackInput)) match {
27           case ScalaFailure(error) =>
28             val msg = s"Failed to deliver message ${message.toDisplay} - ${error.getMessage}"
29             logger.error(msg, error)
30             Some(MessageDeliveryError(message, error))
31           case ScalaSuccess(_) =>
32             val msg = s"Delivered message ${message.toDisplay}"
33             logger.debug(msg)
34             None
35         }
36       } else {
37         val msg = s"Failed to deliver message ${message.toDisplay} - current Rug not available for dependency resolution"
38         logger.error(msg)
39         None
40       }
41     }
42     val instructionResponseFutures: Seq[Future[Iterable[PlanLogEvent]]] = plan.instructions.map { respondable =>
43       val instruction = handleInstruction(plan, respondable, callbackInput)
44       Future {
45         instruction
46       }
47     }
48     val futureInstructionLog: Future[Seq[PlanLogEvent]] = Future.fold(instructionResponseFutures)(Seq[PlanLogEvent]())(_ ++ _)
49     futureInstructionLog.map(instructionLogEvents => PlanResult(messageLog ++ instructionLogEvents))
50   }

52   private def handleInstruction(currentPlan: Plan, plannable: Plannable, callbackInput: Option[Response]): Seq[PlanLogEvent] = {
53     Try {
54       instructionRunner.run(plannable.instruction, callbackInput)
55     } match {
56       case ScalaFailure(error) =>
57         val msg = s"Failed to run ${plannable.toDisplay} - ${error.getMessage}"
58         logger.error(msg, error)
59         Seq(InstructionError(plannable.instruction, error))
60       case ScalaSuccess(response) =>
61         val msg = s"Ran ${plannable.toDisplay} and then ${response.toDisplay}"
62         logger.debug(msg)
63         val callbackOption: Option[Callback] = plannable match {
64           case _: Nonrespondable =>
65             response match {
66               case Response(Success, _, _, Some(plan: Plan)) => Some(plan)
67               case _ => None
68             }
69           case respondable: Respondable =>
70             response match {
71               case Response(Success, _, _, _) => respondable.onSuccess
72               case Response(Failure, _, _, _) => respondable.onFailure
73             }
74         }
75         val callbackResults: Option[Seq[PlanLogEvent]] = callbackOption.map { callback =>
76           Try(handleCallback(currentPlan, callback, Some(response))) match {
77             case ScalaFailure(error) =>
78               val msg = s"Failed to invoke ${callback.toDisplay} after ${plannable.toDisplay} - ${error.getMessage}"
79               logger.error(msg, error)
80               Seq(CallbackError(callback, error))
81             case ScalaSuccess(nestedPlanExecutionOption) =>
82               val msg = s"Invoked ${callback.toDisplay} after ${plannable.toDisplay}"
83               logger.debug(msg)
84               nestedPlanExecutionOption
85           }
86         }
87         //ensure handled errors don't return failure https://github.com/atomist/rug/issues/531
88         (response, callbackResults, callbackOption) match {
89           case (Response(Failure, _, _, _), Some(logs), Some(callback)) if !PlanResultInterpreter.hasLogFailure(logs) && callback.isInstanceOf[Respond] =>
90             val handled = Response(Status.Handled, response.msg, response.code, response.body)
91             Seq(callbackResults, Some(Seq(InstructionResult(plannable.instruction, handled)))).flatten.flatten
92           case _ =>
93             Seq(callbackResults, Some(Seq(InstructionResult(plannable.instruction, response)))).flatten.flatten
94         }
95     }
96   }

98   private def handleCallback(currentPlan: Plan, callback: Callback, instructionResult: Option[Response]): Seq[PlanLogEvent] = callback match {
99     case m: Message =>
100       if (currentPlan.returningRug.nonEmpty) {
101         Try(messageDeliverer.deliver(currentPlan.returningRug.get, m, instructionResult)) match {
102           case ScalaFailure(error) =>
103             val msg = s"Failed to deliver message ${m.toDisplay} - ${error.getMessage}"
104             logger.error(msg, error)
105           case ScalaSuccess(_) =>
106             val msg = s"Delivered message ${m.toDisplay}"
107             logger.debug(msg)
108         }
109       } else {
110         val msg = s"Failed to deliver message ${m.toDisplay} - current Rug not available for dependency resolution"
111         logger.error(msg)
112       }
113       Nil
114     case r: Respond =>
115       handleInstruction(currentPlan, Nonrespondable(r), instructionResult)
116     case p: Plan =>
117       val planResult = runNestedPlan(p, instructionResult)
118       Seq(NestedPlanRun(p, planResult))
119   }
120 
121   private def runNestedPlan(plan: Plan, input: Option[Response] = None) = {
122     val planRunner = nestedPlanRunner.getOrElse(new LocalPlanRunner(messageDeliverer, instructionRunner, nestedPlanRunner, loggerOption))
123     planRunner.run(plan, input)
124   }
