114 private class PairMutableView(

116                                parent: MutableView[_])
117   extends ViewSupport[UpdatableTreeNode](originalBackingObject, parent) {
118 
119   // The backing node has a different name, as the Antlr
120   // grammar doesn't adhere to our capitalization rules
121   require(currentBackingObject.nodeName.equals("pair"))
122 
123   private val kids: Seq[MutableView[_]] =
124     singleChild(currentBackingObject, "value") match {
125       case Some(value: UpdatableTreeNode) =>
126         value.childNodes match {
127           case Seq(_: PaddingTreeNode) =>
128             Seq(new JsonStringView(value, this))
129           case Seq(s: UpdatableTreeNode) if "STRING".equals(s.nodeName) =>
130             Seq(new JsonStringView(s, this))
131           case Seq(mv) if "object" equals mv.nodeName =>
132             findPairsInValueNode(this, mv)
133           case Seq(arr) if "array".equals(arr.nodeName) =>
134             Nil
135           case empty if empty.isEmpty => Nil

137             tooMany.foreach(f => println(TreeNodeUtils.toShortString(f)))

139         }
140       case _ => requiredSingleChild(currentBackingObject, "STRING") match {
141         case s: UpdatableTreeNode if "STRING".equals(s.nodeName) =>
142           Seq(new JsonStringView(s, this))
143       }
144     }
145 
146   override def nodeName: String = nodeNameFromValue(currentBackingObject)
147
