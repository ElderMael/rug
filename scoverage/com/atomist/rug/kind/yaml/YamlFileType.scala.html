56   override def fileToRawNode(f: FileArtifact): Option[PositionedTreeNode] = {
57     // We're currently adding to the node on top of the stack

59     var state: State = InDocument

63     root.setEndPosition(OffsetInputPosition(f.contentLength.toInt))
64     nodeStack.push(root)
65 
66     val reader = new InputStreamReader(f.inputStream())
67     val events = yaml.parse(reader).asScala
68     for (e <- events) {
69       val oldState = state
70       state = state.transition(Input(f.content, e, nodeStack))
71       // println(s"$e from $oldState to $state")
72     }
73     IOUtils.closeQuietly(reader)

86   private case class SeenKey(keyTerminal: String, previousState: State)
87     extends State {
88 
89     protected def on = {
90       case Input(content, s: ScalarEvent, nodeStack) =>
91         // Scalar key with value. Add a container a child of present node and return to Open state
92         val value = scalarToTreeNode(content, s)
93         val container = new YamlPositionedTreeNode(
94           keyTerminal,
95           value.startPosition,
96           value.endPosition,
97           Seq(value),
98           Set(ScalarType),
99           significance = TreeNode.Signal)
100         nodeStack.top.addChild(container)
101         previousState
102       case Input(content, sse: SequenceStartEvent, nodeStack) =>
103         val newContainer = YamlPositionedTreeNode.startAContainer(keyTerminal, markToPosition(content, sse.getStartMark), Set(SequenceType))
104         nodeStack.top.addChild(newContainer)
105         nodeStack.push(newContainer)
106         InSequence(previousState)
107       case Input(content, mse: MappingStartEvent, nodeStack) =>
108         val newContainer = YamlPositionedTreeNode.startAContainer(
109           keyTerminal, markToPosition(content, mse.getStartMark), Set(MappingType))
110         nodeStack.top.addChild(newContainer)
111         nodeStack.push(newContainer)
112         InMapping(previousState)
113     }
114   }
