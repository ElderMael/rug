9 /**

12   * For example, that Repo nodes are tagged "Repo" and have "owner" and "name" keys,

28   override def navigate(context: GraphNode, edgeName: String, executionContext: ExecutionContext): Option[Seq[TreeNode]] = {
29     val repoResolver = executionContext.repoResolver.getOrElse(
30       throw new UnsupportedOperationException(
31         s"No repo resolver available: Passed [$executionContext]")
32     )
33 
34     context match {
35       case repo if repo.hasTag("Repo") =>
36         val owner = NodeUtils.requiredKeyValue(context, "owner")
37         val name = NodeUtils.requiredKeyValue(context, "name")
38         val projectSources =
39           if (isSha(edgeName)) repoResolver.resolveSha(owner, name, edgeName)
40           else repoResolver.resolveBranch(owner, name, edgeName)
41         val pmv = new ProjectMutableView(projectSources)
42         Some(Seq(pmv))
43       case commit if commit.hasTag("Commit") =>
44         // Commit has a sha and associated Repo
45         // We don't care about the edge name, although there should be one
46         val sha = NodeUtils.requiredKeyValue(context, "sha")
47         val (owner, name) = extractOwnerAndRepoFromAssociatedRepo(context)
48         val projectSources = repoResolver.resolveSha(owner, name, sha)
49         val pmv = new ProjectMutableView(projectSources)
50         Some(Seq(pmv))
51       case _ =>
52         None
53     }
54   }
55 
56   private def isSha(s: String) =
57     s.matches("[a-fA-F0-9]{40}")
58 }
