437     (creator, rugResolver) match {
438       case (Some(rug), Some(resolver)) => resolver.resolve(rug, editorName) match {
439         case Some(ed: ProjectEditor) =>
440           ed.modify(currentBackingObject, SimpleParameterValues(params)) match {
441             case sm: SuccessfulModification =>
442               updateTo(sm.result)
443             case _: NoModificationNeeded => currentBackingObject
444             case wtf =>
445               throw new RugRuntimeException(ed.name, s"Unexpected editor failure: $wtf", null)
446           }
447         case None =>
448           if (editorName.contains(":")) {
449             val shortName = StringUtils.substringAfterLast(editorName, ":")
450             if (resolver.resolve(rug, shortName).nonEmpty) {
451               throw new EditorNotFoundException(s"Could not find editor: $editorName. Did you mean: $shortName?")
452             }
453           }
454           throw new EditorNotFoundException(editorName, resolver.resolvedDependencies.rugs.allRugs);
455         case _ =>
456           throw new EditorNotFoundException(editorName, resolver.resolvedDependencies.rugs.allRugs);
457       }
458       case _ => throw new RugRuntimeException(name, s"No Rug context in which to find other editor: [$editorName]")
459     }

517       // Return the path to the file if it's valid, even if we can't resolve a structure within it

519         pathToRootContainer(path, kind)

527     val pexpr = pathToRootContainer(path, kind)
528     ctx.pathExpressionEngine.ee.evaluate(this, pexpr) match {
529       case Right(nodes) if nodes.size == 1 =>

536           case _ => None
537         }
538       case Right(nodes) if nodes.size > 1 =>
539         throw new IllegalArgumentException(s"Found ${nodes.size} hits for [$pexpr], not 1")
540       case x =>
541         //println(s"Unexpected result [$x] for [$pexpr]")
542         None
