43   private def toView(f: FileArtifactBackedMutableView): Option[TreeNode] = {
44     val preprocessedFile = f.currentBackingObject.withContent(preprocess(f.content))
45     val inner = fileToRawNode(preprocessedFile) match {
46       case Some(ptn: ParsedNode) =>
47         Some(TextTreeNodeLifecycle.makeWholeFileNodeReady(name,
48           PositionedTreeNode.fromParsedNode(ptn),

50           preprocess, postprocess))
51       case None => None
52       case x => throw new RuntimeException(s"What is $x")
53     }
54     inner.map(createView(_, f))
55   }
