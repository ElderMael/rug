112   override def toString = {
113     val builder = new StringBuilder

115     _lines.foreach(d => {
116       val lb = "\n".r.findAllMatchIn(d.getRaw).length
117       while (ai.get() < d.getLineno - lb) {

157   private def mergeLines(line: DockerfileLine): Unit = {
158     var _line = line
159 
160     // First get the next element in the order
161     val newCmd = Command.values.find(_.toString == line.name).head
162 
163     var nextCmdLine: DockerfileLine = null
164     var ix: Int = -1
165     for (i <- _lines.indices; if nextCmdLine == null) {
166       val cmd = Command.values.find(_.toString == _lines(i).name).head
167       if (newCmd.id <= cmd.id) {
168         nextCmdLine = _lines(i)
169         ix = i
170       }
171     }
172     if (nextCmdLine != null) {
173       // Insert before
174       val lb = "\n".r.findAllMatchIn(nextCmdLine.getRaw).length
175       _line.setLineno(nextCmdLine.getLineno - lb)
176     } else if (_lines.nonEmpty)
177       _line.setLineno(_lines.last.getLineno + 1)
178     else
179       _line.setLineno(1)
180 
181     if (ix >= 0) {
182       _lines = insertAt(_line, ix, _lines.toList)
183       // Update line numbers in later lines
184       for (i <- (ix + 1) until _lines.size) {
185         _lines(i).setLineno(_lines(i).getLineno + 1)
186       }
187     } else
188       _lines :+= _line
189   }

191   private def insertAt[T](n: T, pos: Int, list: List[T]): List[T] =
192     if (pos >= list.length) {
193       list ::: List(n)
194     } else {
195       list.zipWithIndex.foldLeft(List[T]()) { (acc, tuple) =>
196         tuple match {
197           case (item, idx) =>
198             if (idx == pos) acc ::: List(n, item) else acc ::: List(item)
199         }
200       }
201     }
