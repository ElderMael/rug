42   @ExportFunction(readOnly = false, description = "Rename the given package. All package under it will also be renamed")
43   def renamePackage(@ExportFunctionParameterDescription(
44     name = "oldPackage",
45     description = "Old package name")
46                     oldPackageName: String,
47                     @ExportFunctionParameterDescription(
48                       name = "newPackage",
49                       description = "The new package name")
50                     newPackageName: String): Unit = {
51     // We do this will string operations rather than JavaParser
52     if (!JavaHelpers.isValidPackageName(newPackageName))
53       fail(s"Invalid new package name: [$newPackageName]")
54     val pathToReplace = oldPackageName.replace(".", "/")
55     val newPath = newPackageName.replace(".", "/")
56 
57     // Replace imports and other references, such as annotations and Strings
58     //regexpReplaceWithFilter(isJavaFilter, s"import[\\s]+$oldPackageName", s"import $newPackageName")
59     regexpReplaceWithFilter(mayContainReferencesToClassNames, oldPackageName, newPackageName)
60 
61     regexpReplaceWithFilter(f => f.name.endsWith(JavaExtension) && f.path.contains(pathToReplace),
62       s"package[\\s]+$oldPackageName", s"package $newPackageName")
63 
64     // Fix file structure
65     replaceInPath(pathToReplace, newPath)
66   }
