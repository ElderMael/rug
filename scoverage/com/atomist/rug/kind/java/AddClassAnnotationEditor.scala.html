13 /**
14   * Add a class annotation if it doesn't exist.
15   *
16   * @param selector selects class to annotate
17   * @param annotationPackageName Annotation package None means default package
18   */
19 // TODO this supports only marker annotations: Need superclass to generalize
20 class AddClassAnnotationEditor(selector: ClassOrInterfaceDeclaration => Boolean,
21                                annotationPackageName: Option[String],
22                                annotationName: String,
23                                javaSourcePath: String = JavaConstants.DefaultBaseSourcePath)
24   extends JavaParserProjectEditor("AddClassAnnotation", javaSourcePath)
25     with LazyLogging {
26 
27   private def annotationFqn: String =
28     if (annotationPackageName.isEmpty)
29       annotationName
30     else
31       annotationPackageName.get + "." + annotationName
32 
33   // Could pull into superclass, using Parser
34   override def meetsPostcondition(as: ArtifactSource): Boolean = {
35     val annotatedFiles = as.files
36       .filter(JavaHelpers.isJavaSourceArtifact)
37       .exists(f => {
38         val cu = JavaParser.parse(f.inputStream())
39         cu.getTypes.asScala.exists(_.getAnnotations.asScala.exists(_.getNameAsString.equals(annotationName)))
40       })
41     annotatedFiles
42   }
43 
44   override protected def maybeModifyCompilationUnit(cu: CompilationUnit, poa: ParameterValues): Option[CompilationUnit] = {
45     val modifiedTypes: Traversable[ClassOrInterfaceDeclaration] = cu.getTypes.asScala. collect {
46       case coit: ClassOrInterfaceDeclaration if selector(coit) && JavaTypeType.annotationAddedTo(coit, annotationName) =>
47         coit
48     }
49 
50     if (modifiedTypes.nonEmpty) {
51       if (annotationPackageName.isDefined)
52         JavaParserUtils.addImportsIfNeeded(Seq(annotationFqn), cu)
53       Some(cu)
54     } else
55       None
56   }
57 
58   override def description: String = s"Add @$annotationFqn annotation to class"
