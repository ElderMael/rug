211   private def emitTypes(poa: ParameterValues): Seq[FileArtifact] = {
212     val tsClassOrInterfaces = ListBuffer.empty[StringFileArtifact]
213     val alreadyGenerated = ListBuffer.empty[GeneratedType]
214     val allTypes = typeRegistry.types.sortWith(typeSort)
215     val generatedTypes = allTypes.flatMap(t => t.operations.flatMap(op => getGeneratedTypes(t, op)))
216     val pathParam = poa.stringParamValue(OutputPathParam)
217     val path = if (pathParam.contains("/")) StringUtils.substringBeforeLast(pathParam, "/") + "/" else ""
218 
219     for {
220       t <- generatedTypes
221       if !alreadyGenerated.contains(t)
222     } {
223       val output = new StringBuilder(config.licenseHeader)
224       output ++= config.separator
225       output ++= config.imports
226       output ++= t.specificImports
227       output ++= getImports(generatedTypes, t)
228       output ++= config.separator
229       output ++= s"export { ${t.name} };"
230       output ++= config.separator
231       output ++= t.toString
232       output ++= config.separator
233       tsClassOrInterfaces += StringFileArtifact(s"$path${t.name}.ts", output.toString())
234       alreadyGenerated += t
235     }
236 
237     // Core.ts
238     val output = new StringBuilder(config.licenseHeader)
239     output ++= config.separator
240     val uniqueGeneratedTypes = (alreadyGenerated.groupBy(_.name) map {
241       case (_, l) => l.head
242     }).toSeq.sortBy(_.name)
243     output ++= uniqueGeneratedTypes.map(t => s"""import { ${t.name} } from "./${t.name}";""").mkString("\n")
244     output ++= config.separator
245     output ++= uniqueGeneratedTypes.map(t => s"\nexport { ${t.name} };").mkString("")
246     tsClassOrInterfaces += StringFileArtifact(pathParam, output.toString())
247 
248     tsClassOrInterfaces
249   }

251   private def getImports(interfaceTypes: Seq[GeneratedType], currentType: GeneratedType) = {
252     val imports = interfaceTypes
253       .flatMap(t => currentType.methods.map(m => StringUtils.removeEnd(m.returnType, "[]"))
254         .filter(currentType.name != t.name && _ == t.name))
255       .toList
256 
257     (currentType.parent.toList ::: imports)
258       .distinct
259       .filterNot(_ == currentType.root)
260       .map(i => s"""import { $i } from "./$i";""")
261       .mkString("\n")
262   }
