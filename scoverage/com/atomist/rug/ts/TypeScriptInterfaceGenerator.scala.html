25 /**
26   * Generate interface types for documents.
27   *
28   * @param typeRegistry registry of known Rug Types.
29   */
30 class TypeScriptInterfaceGenerator(typeRegistry: TypeRegistry = DefaultTypeRegistry,
31                                    config: TypeGenerationConfig = TypeGenerationConfig(),
32                                    override val tags: Seq[Tag] = Nil,
33                                    root: String = "TreeNode")
34   extends AbstractTypeScriptGenerator(typeRegistry, config, tags) {
35 
36   private case class InterfaceGeneratedType(name: String,
37                                             description: String,
38                                             methods: Seq[MethodInfo],
39                                             parent: Seq[String] = Seq(root))
40     extends GeneratedType {
41 
42     override def root: String = TypeScriptInterfaceGenerator.this.root
43 
44     override def toString: String = {
45       val output = new StringBuilder
46       output ++= helper.toJsDoc(description)
47       if (parent.isEmpty)
48         output ++= s"\ninterface $name {${config.separator}"
49       else
50         output ++= s"\ninterface $name extends ${parent.mkString(", ")} {${config.separator}"
51 
52       output ++= methods.map(_.toString).mkString(config.separator)
53       output ++= s"${if (methods.isEmpty) "" else config.separator}}${indent.dropRight(1)}"
54       output.toString
55     }
56   }
57 
58   private case class InterfaceMethodInfo(typeName: String,
59                                          name: String,
60                                          params: Seq[MethodParam],
61                                          returnType: String,
62                                          description: Option[String],
63                                          exposeAsProperty: Boolean,
64                                          deprecated: Boolean)
65     extends MethodInfo {
66 
67     override def toString: String = {
68       if (exposeAsProperty) {
69         // Emit property only
70         s"${comment(indent)}${indent}readonly $name: $returnType;"
71       } else {
72         // Emit function
73         s"${comment(indent)}$indent$name(${params.mkString(", ")}): $returnType;"
74       }
75     }
76   }
77 
78   protected def getMethodInfo(typeName: String, op: TypeOperation, params: Seq[MethodParam]): MethodInfo =
79     InterfaceMethodInfo(typeName, op.name, params, helper.rugTypeToTypeScriptType(op.returnType, typeRegistry),
80       Some(op.description), op.exposeAsProperty, op.deprecated)
81 
82   override def getGeneratedTypes(t: Typed, op: TypeOperation): Seq[GeneratedType] = {
83     val generatedTypes = new ListBuffer[GeneratedType]
84     val alreadyAddedMethods = new ListBuffer[MethodInfo]
85 
86     // Get super classes
87     val superClasses = getSuperClasses(op)
88     for (i <- superClasses.indices) {
89       val name = Typed.typeToTypeName(superClasses(i).parent)
90       val parent = if (i == superClasses.size - 1) Seq(root) else Seq(Typed.typeToTypeName(superClasses(i + 1).parent))
91       val methods = superClasses(i).exportedMethods
92       generatedTypes += InterfaceGeneratedType(name, name, methods, parent)
93       alreadyAddedMethods ++= methods
94     }
95 
96     // Get super interfaces
97     val superInterfaces = getSuperInterfaces(op)
98     for (i <- superInterfaces.size to 1 by -1) {
99       val name = Typed.typeToTypeName(superInterfaces(i - 1).parent)
100       val methods = superInterfaces(i - 1).exportedMethods
101       generatedTypes += InterfaceGeneratedType(name, name, methods, Seq())
102       alreadyAddedMethods ++= methods
103     }
104 
105     // Add leaf class
106     val parent =
107       if (superClasses.isEmpty && superInterfaces.isEmpty) Seq(root)
108       else Seq(Typed.typeToTypeName(superClasses.head.parent)) ++ superInterfaces.map(i => Typed.typeToTypeName(i.parent))
109 
110     val methods = allMethods(t.name, t.operations).filterNot(alreadyAddedMethods.contains(_))
111     generatedTypes += InterfaceGeneratedType(t.name, t.description, methods, parent)
112 
113     generatedTypes
114   }
