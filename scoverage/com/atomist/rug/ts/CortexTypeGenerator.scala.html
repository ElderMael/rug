51   /**
52     * Extract the types described in this document and output a valid node module
53     */
54   def extract(json: String): Set[Typed] = {
55     require(json != null)
56     val doc: Map[String, List[_]] = mapper.readValue(json, classOf[Map[String, List[_]]])
57 
58     def defineProperty(propName: String, typ: String): Prop = {
59       val genTyp = typ match {
60         case "string" => "String"
61         case "number" => "long"
62         case "url" => "String" // TODO we should somehow use helpers for this and construct in the proxy from a string
63         case "timestamp" => "String"
64         case x => x
65       }
66       SimpleProp(propName, genTyp)
67     }
68 
69     /*
70     {"nodes":
71     {"labels":["Push"],
72     "properties":
73     [["before", "string"], ["after", "string"],
74     ["timestamp", "timestamp"], ["branch", "string"]],
75     "unique":["branch", "after"]},
76      */
77     val propertyNodes: Traversable[PropertyNode] =
78       doc("nodes").map(node => {
79         // A node is of form labels, properties, unique?
80         val n = node.asInstanceOf[Map[String, _]]
81         val props: Seq[Prop] = n("properties") match {
82           case l: List[_] =>
83             l.map {
84               case List(id: String, typ: String) =>
85                 defineProperty(id, typ)
86               case List(id: String, legalValues: List[String]@unchecked) =>
87                 EnumProp(id, legalValues)
88             }
89         }
90 
91         val pn = PropertyNode(
92           n("labels").asInstanceOf[List[String]].toSet,
93           props
94         )
95         pn
96       })
97 
98     // These are normally a - [right,cardinality] - right
99     // ["Org", ["HAS", ["repo", "1:M"], ["org", "1:1"]], "Repo"]
100     // List(ChatTeam, List(OWNS, List(orgs, 1:M), List(chatTeam, 1:1)), Org)
101     val allRelationships: Traversable[Relationship] =
102     doc.getOrElse("relationships", Nil) flatMap {
103       case List(leftEntity: String, List(_: String, List(lrName: String, lrCard: String), List(rlName: String, rlCard: String)), rightEntity: String) =>
104         Seq(
105           Relationship(leftEntity, lrName, Cardinality(lrCard), rightEntity),
106           Relationship(rightEntity, rlName, Cardinality(rlCard), leftEntity)
107         )
108       case List(leftEntity: String, List(discard: String, List(lrName: String, lrCard: String)), rightEntity: String) =>
109         Seq(
110           Relationship(leftEntity, lrName, Cardinality(lrCard), rightEntity)
111         )
112       case x =>
113         throw new IllegalArgumentException(s"Illegal list, not able to destructure, $x")
114     }
115 
116     propertyNodes.map(pn => new JsonBackedTyped(pn, allRelationships)).toSet
117   }

144 private class JsonBackedTyped(
145                                properties: PropertyNode,
146                                allRelationships: Traversable[Relationship]
147                              ) extends Typed {
148 
149   override val name: String = properties.labels.head
150 
151   override def description: String = s"Type $name"
152 
153   // TODO should come from parent?
154   override def allOperations: Seq[TypeOperation] = operations
155 
156   override def operations: Seq[TypeOperation] = {
157     val relOps: Seq[TypeOperation] =
158       allRelationships
159         .filter(_.left == name)
160         .map(rel => TypeOperation(
161           name = rel.name,
162           description = s"${rel.name} - ${rel.left} -> ${rel.right}",
163           readOnly = false,
164           parameters = Nil,
165           returnType = rel.cardinality match {
166             case OneToOne => SimpleParameterOrReturnType(rel.right, isArray = false)
167             case OneToM => SimpleParameterOrReturnType(rel.right, isArray = true)
168           },
169           definedOn = null,
170           example = None,
171           exposeAsProperty = true
172         )).toSeq
173     val propsOps: Seq[TypeOperation] =
174       properties.properties
175         // Filter out properties that have the same name as a relationship, that takes precedence
176         .filterNot(p => relOps.exists(_.name == p.name))
177         .map(prop =>
178           TypeOperation(
179             name = prop.name,
180             description = prop.description,
181             readOnly = false,
182             parameters = Nil,
183             returnType = prop.toType,
184             definedOn = null,
185             example = None,
186             exposeAsProperty = true
187           ))
188 
189     propsOps ++ relOps
190   }
191 
192   override def toString: String =
193     s"$name: ops=[${operations.mkString(",")}]"
