34   def editor(name: String): ProjectEditor = {
35     rugs match {
36       case Some(r) =>
37         r.editors.find(e => e.name == name) match {
38           case Some(e) => e
39           case _ => throw new RugNotFoundException(
40             s"Editor with name '$name' can not be found in current context. Known editors are [${r.editorNames.mkString(", ")}]")

42       case _ => throw new RugNotFoundException("No context provided")
43     }

85   def editWith(editor: ProjectEditor, params: Any): Unit = {
86     import scala.util.control.Exception._
87 
88     val editorResult = allCatch.either(editor.modify(project.currentBackingObject, parameters(params)))
89     editorResults = editorResults :+ editorResult
90     editorResult match {
91       case Right(sm: SuccessfulModification) =>
92         project.updateTo(sm.result)
93       case Right(_) =>
94       // We've already logged it. Do nothing
95       case Left(ipe: InvalidParametersException) =>
96         throw ipe
97       case Left(unknown) =>
98         throw unknown
99     }
100   }

102   def editWith(editor: ProjectEditor): Unit = {
