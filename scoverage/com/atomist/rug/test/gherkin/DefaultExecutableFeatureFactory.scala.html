9 /**
10   * Default implementation of ExecutableFeatureFactory, which
11   * knows about project features and handler features (coming soon), which it identifies
12   * by their respective subdirectories under .atomist/test.
13   * Using location to determine type avoids the need to make the user
14   * provide additional metadata in their test definitions.
15   */
16 object DefaultExecutableFeatureFactory extends ExecutableFeatureFactory {
17 
18   private val atomistConfig = DefaultAtomistConfig
19 
20   override def executableFeatureFor(f: FeatureDefinition, definitions: Definitions, rugAs: ArtifactSource, rugs: Option[Rugs], listeners: Seq[GherkinExecutionListener], config: GherkinRunnerConfig): AbstractExecutableFeature[_] = {
21     // TODO clean up name of test directory
22     if (f.definition.path.contains(s"${atomistConfig.testsDirectory}/project"))
23       new ProjectManipulationFeature(f, definitions, rugAs, rugs, listeners, config)
24     else if (f.definition.path.contains(s"${atomistConfig.testsDirectory}/${atomistConfig.handlersDirectory}/command"))
25       new CommandHandlerFeature(f, definitions, rugAs, rugs, listeners, config)
26     else if (f.definition.path.contains(s"${atomistConfig.testsDirectory}/${atomistConfig.handlersDirectory}/event"))
27       new EventHandlerFeature(f, definitions, rugAs, rugs, listeners, config)
28     else {
29       throw new IllegalArgumentException(s"Cannot handle path [${f.definition.path}]: Paths must be of form [${atomistConfig.testsDirectory}/project] or [${atomistConfig.testsDirectory}/handlers]")
30     }
31   }
32 }
