44   private def findMatch(s: String): Option[StepMatch] = {
45     // Find a key that matches the regex, then extract the capture groups
46     stepRegistry.keys.flatMap(key => {
47       val r = s.matches(key)
48       if (r) {
49         val om: Match = key.r.findPrefixMatchOf(s).get
50         val args: Seq[AnyRef] = om.subgroups
51         Some(StepMatch(stepRegistry(key), args))
52       }
53       else None
54     }).headOption
55   }
