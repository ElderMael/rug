37   private def executeScenario(scenario: ScenarioDefinition): ScenarioResult = {
38     listeners.foreach(_.scenarioStarting(scenario))
39     val world = createWorldForScenario()
40     val assertionResults: Seq[AssertionResult] =
41       scenario.getSteps.asScala.flatMap(step => {
42         listeners.foreach(_.stepStarting(step))
43         val result = step.getKeyword match {
44           case "Given " if !world.aborted =>
45             runGiven(world, step)
46           case "When " if !world.aborted =>
47             runWhen(world, step)
48           case "Then " if step.getText == "the scenario aborted" =>
49             Some(AssertionResult(step.getText, Result(world.aborted, "Scenario aborted")))
50           case "Then " if !world.aborted =>
51             Some(runThen(world, step))
52           case "Then " if world.aborted =>
53             Some(AssertionResult(step.getText,
54               Failed(s"Scenario aborted: Could not evaluate: [${world.abortMessage}]")))
55           case _ =>
56             None
57         }
58         listeners.foreach(_.stepCompleted(step))
59         result
60       })
61     val sr = ScenarioResult(scenario, assertionResults, "")
62     listeners.foreach(_.scenarioCompleted(scenario, sr))
63     sr
64   }

66   private def runThen(world: ScenarioWorld, step: Step): AssertionResult = {
67     val somo = definitions.thenFor(step.getText)
68     logger.debug(s"Then for [${step.getText}]=$somo")
69     somo match {
70       case Some(stepMatch) =>
71         val r = callFunction(stepMatch, world)
72         r match {
73           case Right(b: java.lang.Boolean) =>
74             AssertionResult(step.getText, Result(b, stepMatch.jsVar.toString))
75           case Right(rsom: ScriptObjectMirror) =>
76             val result = NashornUtils.stringProperty(rsom, "result", "false") == "true"
77             AssertionResult(step.getText, Result(result, NashornUtils.stringProperty(rsom, "message", "Detailed information unavailable")))
78           case Right(r) if ScriptObjectMirror.isUndefined(r) =>
79             // Returning void (which will be undefined) is truthy
80             // This enables use of frameworks such as as chai
81             AssertionResult(step.getText, Result(f = true, stepMatch.jsVar.toString))
82           case Right(wtf) =>
83             throw new IllegalArgumentException(s"Unexpected result from Then '${step.getText}': $wtf")
84           case Left(t) =>
85             AssertionResult(step.getText, Failed(t.getMessage))
86         }
87       case None =>
88         AssertionResult(step.getText, NotYetImplemented("Then " + step.getText))
89     }
90   }

92   private def runWhen(world: ScenarioWorld, step: Step): Option[AssertionResult] = {
93     val somo = definitions.whenFor(step.getText)
94     logger.debug(s"When for [${step.getText}]=$somo")
95     somo match {
96       case Some(som) =>
97         callFunction(som, world) match {
98           case Left(ipe: InvalidParametersException) =>
99             listeners.foreach(_.stepFailed(step, ipe))
100             world.logInvalidParameters(ipe)
101             None
102           case Left(e: Exception) if e.getCause != null && e.getCause.isInstanceOf[InvalidParametersException] =>
103             // Can sometimes get this wrapped
104             listeners.foreach(_.stepFailed(step, e.getCause.asInstanceOf[InvalidParametersException]))
105             world.logInvalidParameters(e.getCause.asInstanceOf[InvalidParametersException])
106             None
107           case Left(t) =>
108             listeners.foreach(_.stepFailed(step, t))
109             logger.error(t.getMessage, t)
110             world.abort(t.getMessage)
111             None
112           case _ =>
113             None
114         }
115       case None =>
116         Some(AssertionResult(step.getText, NotYetImplemented("When " + step.getText)))
117     }
118   }

133         }

136     }

139   // Call a ScriptObjectMirror function with appropriate error handling
140   private def callFunction(sm: StepMatch, world: ScenarioWorld): Either[Throwable, Object] = {
141     import scala.util.control.Exception._
142     val target = world.target match {
143       case gn: GraphNode => new jsSafeCommittingProxy(gn, world.typeRegistry)
144       case t => t
145     }
146     // Only include the target if it's different from the world.
147     val fixedParams: Seq[AnyRef] = target match {
148       case `world` => Seq(world)
149       case _ => Seq(target, world)
150     }
151     val args = fixedParams ++ sm.args
152     allCatch.either(sm.jsVar.call("apply", args:_*))
153   }
