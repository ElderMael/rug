131   def planIsInternallyValid(): Boolean = {
132     if (recordedPlans.isEmpty)
133       throw new IllegalArgumentException("No plan was recorded")
134     recordedPlans.values.forall(p => {
135       !p.instructions.map(_.instruction).exists {
136         case Edit(detail) =>
137           val knownEditors: Seq[String] = rugs.map(_.editorNames).getOrElse(Nil)
138           !knownEditors.contains(detail.name)
139         case Generate(detail) =>
140           val knownGenerators: Seq[String] = rugs.map(_.generatorNames).getOrElse(Nil)
141           !knownGenerators.contains(detail.name)
142         case Command(detail) =>
143           val knownCommandHandlers: Seq[String] = rugs.map(_.commandHandlerNames).getOrElse(Nil)
144           !knownCommandHandlers.contains(detail.name)
145         // TODO there are probably more cases here
146         case _ => false
147       }
148     })
149   }
