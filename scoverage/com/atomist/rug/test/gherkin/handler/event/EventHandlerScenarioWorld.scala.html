25   def eventHandler(name: String): EventHandler = {
26     rugs match {
27       case Some(r) =>
28         r.eventHandlers.find(e => e.name == name) match {
29           case Some(e) => e
30           case _ => throw new RugNotFoundException(
31             s"EventHandler with name '$name' can not be found in current context. Known EventHandlers are [${r.eventHandlerNames.mkString(", ")}]")
32         }
33       case _ => throw new RugNotFoundException("No context provided")
34     }
35   }
36 
37   def registerHandler(name: String): EventHandler = {
38     val eh = eventHandler(name)
39     registeredHandlers.append(eh)
40     eh
41   }

43   /**
44     * Publish an event constructed in TypeScript using our API
45     * (normally a cortex stub)
46     * It's hopefully a ScriptObjectMirror
47     */
48   def sendEvent(e: AnyRef): Unit = {
49     val gn = NashornMapBackedGraphNode.toGraphNode(e).getOrElse(
50       throw new IllegalArgumentException(s"Cannot make a GraphNode out of $e")
51     )
52     if (registeredHandlers.isEmpty)
53       throw new IllegalStateException("No handler is registered")
54     for (h <- registeredHandlers) {
55       handleEventNode(gn, h)
56     }
57   }

59   private def handleEventNode(eventNode: GraphNode, h: EventHandler) = {
60     if (eventNode.nodeTags.contains(h.rootNodeName)) {
61       val tm = new TreeMaterializer {
62         override def rootNodeFor(e: SystemEvent, pe: PathExpression) = eventNode
63 
64         override def hydrate(teamId: String, rawRootNode: GraphNode, pe: PathExpression) = rawRootNode
65       }
66       val rugContext: RugContext = createRugContext(tm)
67 
68       // Check if it matches, if we can
69       notifyListenersOfMatchResult(eventNode, h, rugContext)
70 
71       //println("About to handle event")
72       val plan = h.handle(rugContext, SystemEvent(rugContext.teamId, h.rootNodeName, 1))
73       plan.foreach(recordPlan(h.name, _))
74     }
75     else {
76       notifyListenersOfMatchPossible(eventNode, h)
77     }
78   }
79 
80   private def notifyListenersOfMatchResult(eventNode: GraphNode, h: EventHandler, rugContext: RugContext) = {
81     h match {
82       case jsh: JavaScriptEventHandler =>
83         rugContext.pathExpressionEngine.ee.evaluate(JavaScriptEventHandler.rootNodeFor(eventNode), jsh.pathExpression, rugContext) match {
84           case Right(nodes) =>
85             //println(s"Results for [${jsh.pathExpressionStr}] were ${nodes}")
86             for (l <- listeners)
87               l.pathExpressionResult(PathExpressionEvaluation(jsh.pathExpressionStr, eventNode, nodes))
88           case Left(_) =>
89           // The evaluation failed. Who cares. The test will blow up anyway
90         }
91       case _ =>
92       // We can't find the path expression to check the match for
93     }
94   }
95 
96   private def notifyListenersOfMatchPossible(eventNode: GraphNode, h: EventHandler) = {
97     h match {
98       case jsh: JavaScriptEventHandler =>
99         for (l <- listeners)
100           l.pathExpressionResult(PathExpressionEvaluation(jsh.pathExpressionStr, eventNode, Nil))
101       case _ =>
102       // We can't find the path expression to check the match for
103     }
104   }
